        function waitForGlobal(name, timeoutMs) {
          return new Promise((resolve, reject) => {
            if (window[name]) return resolve(window[name]);
            const deadline = Date.now() + (timeoutMs || 5000);
            const timer = setInterval(() => {
              if (window[name]) {
                clearInterval(timer);
                resolve(window[name]);
              } else if (Date.now() > deadline) {
                clearInterval(timer);
                reject(new Error(`${name} did not load`));
              }
            }, 50);
          });
        }

        function isSingleSvgMarkup(html) {
          if (!html || !/<svg[\s>]/i.test(html)) return false;
          const tmp = document.createElement("div");
          tmp.innerHTML = html;
          if (
            tmp.children.length !== 1 ||
            tmp.firstElementChild.tagName.toLowerCase() !== "svg"
          ) {
            return false;
          }
          // Only stray text *around* the SVG disqualifies it -- text inside
          // the SVG (e.g. a <text> label like "r = 4 cm") is legitimate
          // diagram content.
          for (const node of tmp.childNodes) {
            if (node.nodeType === 3 && (node.nodeValue || "").trim() !== "") {
              return false;
            }
          }
          return true;
        }

        function buildFnGraphBlotContent(node) {
          node.innerHTML = "";
          const label = document.createElement("span");
          label.className = "ql-fn-graph-label";
          label.textContent = "Function graph (double-click to edit)";
          const preview = document.createElement("span");
          preview.className = "ql-fn-graph-preview";
          node.appendChild(label);
          node.appendChild(preview);

          const spec = normalizeGraphSpec(node.getAttribute("data-graph"));
          if (!spec) {
            preview.textContent = "Invalid graph settings";
            return;
          }
          if (typeof window.functionPlot === "function") {
            try {
              window.functionPlot(
                graphSpecToPlotOptions(spec, preview, 280),
              );
              appendGraphLegend(preview, spec);
              return;
            } catch {}
          }
          preview.textContent = spec.functions
            .map((f) => f.label || f.fn)
            .join(" · ");
        }

        function renderLatexNode(node, latex, displayMode) {
          while (node.firstChild) {
            node.removeChild(node.firstChild);
          }
          if (window.katex) {
            const wrap = document.createElement("span");
            try {
              window.katex.render(latex, wrap, {
                throwOnError: false,
                displayMode,
                output: "html",
              });
              node.appendChild(wrap);
              return;
            } catch {}
          }
          node.textContent = displayMode
            ? `$$${latex}$$`
            : `\\(${latex}\\)`;
        }

        function loadHtmlIntoQuill(quill, html) {

          quill.setContents([], "silent");
          quill.clipboard.dangerouslyPasteHTML(0, safe, "silent");
        }

        function preprocessHtmlForQuill(html) {
          const parser = new DOMParser();
          const doc = parser.parseFromString(
            `<div id="__root">${html}</div>`,
            "text/html",
          );
          const root = doc.getElementById("__root");
          if (!root) return html;

          // Convert graph embeds first so a rendered plot's <svg> is never
          // mis-wrapped as a raw-html diagram below.
          root.querySelectorAll(".fn-graph").forEach((el) => {
            const div = doc.createElement("div");
            div.className = "ql-fn-graph";
            div.setAttribute("data-graph", el.getAttribute("data-graph") || "");
            el.replaceWith(div);
          });

          root.querySelectorAll("svg").forEach((svg) => {
            const wrap = doc.createElement("div");
            wrap.className = "ql-raw-html";
            wrap.setAttribute(
              "data-raw-html",
              encodeURIComponent(svg.outerHTML),
            );
            wrap.setAttribute("data-raw-type", "svg");
            const label = doc.createElement("span");
            label.className = "ql-raw-html-label";
            label.textContent = "Diagram (double-click to edit)";
            const content = doc.createElement("span");
            content.className = "ql-raw-html-content";
            content.innerHTML = svg.outerHTML;
            wrap.appendChild(label);
            wrap.appendChild(content);
            svg.replaceWith(wrap);
          });

          const walker = doc.createTreeWalker(root, NodeFilter.SHOW_TEXT);
          const textNodes = [];
          let n;
          while ((n = walker.nextNode())) textNodes.push(n);
          const latexRegex =
            /\$\$([\s\S]+?)\$\$|\\\(([\s\S]+?)\\\)|\\\[([\s\S]+?)\\\]/g;
          textNodes.forEach((textNode) => {
            const text = textNode.nodeValue || "";
            if (!/\$\$|\\\(|\\\[/.test(text)) return;
            latexRegex.lastIndex = 0;
            const parts = [];
            let last = 0;
            let m;
            while ((m = latexRegex.exec(text)) !== null) {
              if (m.index > last) {
                parts.push({ type: "text", value: text.slice(last, m.index) });
              }
              if (m[1] !== undefined) {
                parts.push({ type: "block", value: m[1] });
              } else if (m[2] !== undefined) {
                parts.push({ type: "inline", value: m[2] });
              } else if (m[3] !== undefined) {
                parts.push({ type: "block", value: m[3] });
              }
              last = latexRegex.lastIndex;
            }
            if (parts.length === 0) return;
            if (last < text.length) {
              parts.push({ type: "text", value: text.slice(last) });
            }
            const frag = doc.createDocumentFragment();
            parts.forEach((part) => {
              if (part.type === "text") {
                frag.appendChild(doc.createTextNode(part.value));
              } else if (part.type === "inline") {
                const span = doc.createElement("span");
                span.className = "ql-latex-inline";
                span.setAttribute("data-latex", part.value);
                span.textContent = `\\(${part.value}\\)`;
                frag.appendChild(span);
              } else {
                const div = doc.createElement("div");
                div.className = "ql-latex-block";
                div.setAttribute("data-latex", part.value);
                div.textContent = `$$${part.value}$$`;
                frag.appendChild(div);
              }
            });
            textNode.replaceWith(frag);
          });

          return root.innerHTML;
        }

        function getHtmlFromQuill(quill) {
          const clone = quill.root.cloneNode(true);

          clone.querySelectorAll(".ql-latex-inline").forEach((el) => {
            const latex = el.getAttribute("data-latex") || "";
            el.replaceWith(document.createTextNode(`\\(${latex}\\)`));
          });
          clone.querySelectorAll(".ql-latex-block").forEach((el) => {
            const latex = el.getAttribute("data-latex") || "";
            el.replaceWith(document.createTextNode(`$$${latex}$$`));
          });
          clone.querySelectorAll(".ql-fn-graph").forEach((el) => {
            const div = document.createElement("div");
            div.className = "fn-graph";
            div.setAttribute("data-graph", el.getAttribute("data-graph") || "");
            el.replaceWith(div);
          });
          clone.querySelectorAll(".ql-raw-html").forEach((el) => {
            const enc = el.getAttribute("data-raw-html");
            let raw = "";
            try {
              raw = enc ? decodeURIComponent(enc) : "";
            } catch {
              raw = "";
            }
            const tmp = document.createElement("div");
            tmp.innerHTML = raw;
            const nodes = Array.from(tmp.childNodes);
            if (nodes.length) {
              const first = nodes[0];
              el.replaceWith(first);
              let prev = first;
              for (let i = 1; i < nodes.length; i++) {
                prev.parentNode.insertBefore(nodes[i], prev.nextSibling);
                prev = nodes[i];
              }
            } else {
              el.remove();
            }
          });

          clone.querySelectorAll("p").forEach((p) => {
            if (p.innerHTML.trim() === "<br>") {
              p.innerHTML = "";
            }
          });

          let html = clone.innerHTML;
          html = html.replace(/<p>\s*<\/p>/g, "");
          return html.trim();
        }

        function insertEmbedAtCursor(blotName, value) {
          if (!state.quillInstance || !value) return;
          const range = state.quillInstance.getSelection(true) || {
            index: state.quillInstance.getLength(),
            length: 0,
          };
          state.quillInstance.insertEmbed(range.index, blotName, value, "user");
          state.quillInstance.setSelection(
            range.index + 1,
            0,
            "silent",
          );
        }

                  wrap_line_length: 100,
                  preserve_newlines: false,
                  max_preserve_newlines: 1,
                  end_with_newline: false,

                    "big",
                    "i",
                    "small",
                    "tt",
                    "abbr",
                    "acronym",
                    "cite",

        function normalizeGraphSpec(raw) {
          let spec = raw;
          if (typeof raw === "string") {
            try {
              spec = JSON.parse(raw);
            } catch {
              return null;
            }
          }
          if (!spec || typeof spec !== "object") return null;

          const fns = Array.isArray(spec.functions) ? spec.functions : [];
          const functions = fns
            .filter((f) => f && typeof f.fn === "string" && f.fn.trim())
            .map((f, i) => {
              const out = {
                fn: f.fn.trim(),
                color:
                  typeof f.color === "string" && f.color.trim()
                    ? f.color.trim()
                    : GRAPH_COLOR_CYCLE[i % GRAPH_COLOR_CYCLE.length],
              };
              if (typeof f.label === "string" && f.label.trim()) {
                out.label = f.label.trim();
              }
              const samples = Number(f.nSamples);
              if (Number.isFinite(samples) && samples > 0) {
                out.nSamples = clamp(Math.round(samples), 50, 2000);
              }
              if (f.closed === true) {
                out.closed = true;
              }
              return out;
            });
          if (!functions.length) return null;

          const readAxis = (axis, fallbackLabel) => {
            const out = {
              label:
                axis && typeof axis.label === "string"
                  ? axis.label
                  : fallbackLabel,
            };
            if (
              axis &&
              Array.isArray(axis.domain) &&
              axis.domain.length === 2 &&
              axis.domain.every((v) => Number.isFinite(v)) &&
              axis.domain[0] < axis.domain[1]
            ) {
              out.domain = [axis.domain[0], axis.domain[1]];
            }
            return out;
          };

          const normalized = {
            width: clamp(Math.round(Number(spec.width) || 360), 120, 680),
            height: clamp(Math.round(Number(spec.height) || 280), 100, 600),
            grid: spec.grid !== false,
            xAxis: readAxis(spec.xAxis, "x"),
            yAxis: readAxis(spec.yAxis, "y"),
            functions,
          };
          if (!normalized.xAxis.domain) {
            normalized.xAxis.domain = [-7, 7];
          }
          if (typeof spec.title === "string" && spec.title.trim()) {
            normalized.title = spec.title.trim();
          }
          return normalized;
        }

        function graphSpecToPlotOptions(spec, target, maxWidth) {
          const scale =
            maxWidth && spec.width > maxWidth ? maxWidth / spec.width : 1;
          const options = {
            target,
            width: Math.round(spec.width * scale),
            height: Math.round(spec.height * scale),
            grid: spec.grid,
            disableZoom: true,
            xAxis: { ...spec.xAxis },
            yAxis: { ...spec.yAxis },
            data: spec.functions.map((f) => {
              const d = { fn: f.fn, color: f.color, graphType: "polyline" };
              if (f.nSamples) d.nSamples = f.nSamples;
              if (f.closed) d.closed = true;
              return d;
            }),
          };
          if (spec.title) options.title = spec.title;
          return options;
        }

        function renderFunctionGraph(div) {
          div.innerHTML = "";
          div.removeAttribute("data-graph-error");
          const spec = normalizeGraphSpec(div.getAttribute("data-graph"));
          if (!spec || typeof window.functionPlot !== "function") {
            div.setAttribute("data-graph-error", "true");
            return;
          }
          try {
            window.functionPlot(graphSpecToPlotOptions(spec, div));
            appendGraphLegend(div, spec);
          } catch (err) {
            console.error("Graph render failed:", err);
            div.innerHTML = "";
            div.setAttribute("data-graph-error", "true");
          }
        }

            item.appendChild(swatch);

            legend.appendChild(item);

        }

        function toHexColor(color) {
          if (/^#[0-9a-fA-F]{6}$/.test(color)) return color;
          const probe = document.createElement("span");
          probe.style.color = color;
          document.body.appendChild(probe);
          const computed = getComputedStyle(probe).color;
          probe.remove();
          const m = computed.match(/rgba?\((\d+),\s*(\d+),\s*(\d+)/);
          if (!m) return "#0f766e";
          return (
            "#" +
            [m[1], m[2], m[3]]
              .map((v) => Number(v).toString(16).padStart(2, "0"))
              .join("")
          );
        }

        function collectGraphSpecFromForm(dialog) {
          const functions = Array.from(
            dialog.querySelectorAll(".graph-fn-row"),
          )
            .map((row) => ({
              fn: row.querySelector(".graph-fn-expr").value.trim(),
              color: row.querySelector(".graph-fn-color").value,
              label: row.querySelector(".graph-fn-label").value.trim(),
            }))
            .filter((f) => f.fn);

          const num = (id) => {
            const value = dialog.querySelector(`#${id}`).value.trim();
            return value === "" ? null : Number(value);
          };

          const spec = {
            width: num("graph-width") || 360,
            height: num("graph-height") || 280,
            grid: dialog.querySelector("#graph-grid").checked,
            xAxis: {
              label: dialog.querySelector("#graph-x-label").value.trim() || "x",
            },
            yAxis: {
              label: dialog.querySelector("#graph-y-label").value.trim() || "y",
            },
            functions,
          };
          const xMin = num("graph-x-min");
          const xMax = num("graph-x-max");
          if (xMin !== null && xMax !== null) {
            spec.xAxis.domain = [xMin, xMax];
          }
          if (!dialog.querySelector("#graph-y-auto").checked) {
            const yMin = num("graph-y-min");
            const yMax = num("graph-y-max");
            if (yMin !== null && yMax !== null) {
              spec.yAxis.domain = [yMin, yMax];
            }
          }
          return normalizeGraphSpec(spec);
        }

        function sanitizeSvgString(str) {
          if (!str || !str.trim()) return null;
          const doc = new DOMParser().parseFromString(str, "text/html");
          const svg = doc.querySelector("svg");
          if (!svg) return null;
          svg.querySelectorAll("script").forEach((n) => n.remove());
          const stripEditorAttrs = (el) => {
            Array.from(el.attributes).forEach((attr) => {
              if (
                attr.name.startsWith("se:") ||
                attr.name.startsWith("xmlns:se")
              ) {
                el.removeAttribute(attr.name);
              }
            });
          };
          stripEditorAttrs(svg);

          function applyViewBoxAttrs() {

            svgEl.setAttribute("width", String(Math.round(vb.w * zoom)));

          function resizeCanvas(dw, dh) {
            const nw = Math.max(canvasMin, Math.min(canvasMax, vb.w + dw));
            const nh = Math.max(canvasMin, Math.min(canvasMax, vb.h + dh));
            if (nw === vb.w && nh === vb.h) return;
            vb = { ...vb, w: nw, h: nh };
            applyViewBoxAttrs();
            renderHandles();
            commit();

          function toUserCoords(evt) {
            const pt = svgEl.createSVGPoint();
            pt.x = evt.clientX;
            pt.y = evt.clientY;
            const ctm = svgEl.getScreenCTM();
            if (!ctm) return { x: 0, y: 0 };
            const u = pt.matrixTransform(ctm.inverse());
            return { x: u.x, y: u.y };
          }

          function dashArrayFor(styleName, strokeWidth) {
            const w = Math.max(1, strokeWidth || 1);
            if (styleName === "dashed") return `${w * 3} ${w * 2}`;
            if (styleName === "dotted") return `${w} ${w * 2}`;
            return null;
          }

          function applyStyles(el) {
            const tag = el.tagName.toLowerCase();
            if (tag === "text") {
              // Font-size is managed exclusively by the font-size dropdown
              // so changing stroke/fill on a mixed selection doesn't reset
              // text sizes the user already chose.
              el.setAttribute("fill", style.stroke);
              el.removeAttribute("stroke");
              return;
            }
            el.setAttribute("stroke", style.stroke);
            el.setAttribute("stroke-width", String(style.strokeWidth));
            if (tag === "line" || tag === "polyline") {
              el.setAttribute("fill", "none");
            } else {
              el.setAttribute("fill", style.fill);
            }
            const da = dashArrayFor(style.dashStyle, style.strokeWidth);
            if (da) {
              el.setAttribute("stroke-dasharray", da);
            } else {
              el.removeAttribute("stroke-dasharray");
            }
          }

          function getAngle(el) {
            const a = Number(el && el.dataset.mseAngle);
            return Number.isFinite(a) ? a : 0;
          }

          function bboxOf(el) {
            try {
              return el.getBBox();
            } catch (err) {
              return null;
            }
          }

          function bboxCenter(el) {
            const b = bboxOf(el);
            if (!b) return null;
            return { x: b.x + b.width / 2, y: b.y + b.height / 2 };
          }

          function setRotation(el, angle, pivot) {
            const c = pivot || bboxCenter(el);
            if (!c) return;
            const a = Number(angle) || 0;
            el.dataset.mseAngle = String(a);
            if (a === 0) {
              el.removeAttribute("transform");
            } else {
              el.setAttribute("transform", `rotate(${a} ${c.x} ${c.y})`);
            }
          }

          function refreshPivot(el) {
            const a = getAngle(el);
            if (a === 0) {
              el.removeAttribute("transform");
              return;
            }
            const c = bboxCenter(el);
            if (!c) return;
            el.setAttribute("transform", `rotate(${a} ${c.x} ${c.y})`);
          }

          function ensureArrowMarker(kind) {
            const id = `mse-arrow-${kind}-${instanceId}`;
            if (defsEl.querySelector(`#${id}`)) return id;
            const marker = document.createElementNS(SVG_NS, "marker");
            marker.setAttribute("id", id);
            marker.setAttribute("viewBox", "0 0 10 10");
            marker.setAttribute("refX", "9");
            marker.setAttribute("refY", "5");
            marker.setAttribute("markerWidth", "5");
            marker.setAttribute("markerHeight", "5");
            marker.setAttribute("orient", "auto-start-reverse");
            marker.setAttribute("markerUnits", "strokeWidth");
            const path = document.createElementNS(SVG_NS, "path");
            if (kind === "open") {
              path.setAttribute("d", "M0,1 L10,5 L0,9");
              path.setAttribute("fill", "none");
              path.setAttribute("stroke", "context-stroke");
              path.setAttribute("stroke-width", "1.6");
              path.setAttribute("stroke-linecap", "round");
              path.setAttribute("stroke-linejoin", "round");
            } else {
              path.setAttribute("d", "M0,0 L10,5 L0,10 z");
              path.setAttribute("fill", "context-stroke");
            }
            marker.appendChild(path);
            defsEl.appendChild(marker);
            return id;
          }

          function setLineArrowStyle(el, styleId) {
            el.removeAttribute("marker-start");
            el.removeAttribute("marker-end");
            if (!styleId) return;
            const kind = styleId.startsWith("solid") ? "solid" : "open";
            const id = ensureArrowMarker(kind);
            el.setAttribute("marker-end", `url(#${id})`);
            if (styleId.endsWith("both")) {
              el.setAttribute("marker-start", `url(#${id})`);
            }
          }

          function getLineArrowStyle(el) {
            const ms = el.getAttribute("marker-start") || "";
            const me = el.getAttribute("marker-end") || "";
            if (!me) return null;
            const kind = me.includes("solid") ? "solid" : "open";
            return ms ? `${kind}-both` : `${kind}-end`;
          }

          function pickArrow(styleId) {
            if (!selected || selected.tagName.toLowerCase() !== "line") return;
            const current = getLineArrowStyle(selected);
            setLineArrowStyle(selected, current === styleId ? null : styleId);
            updateSelectionDependentUi();
            commit();
          }

          function refreshArrowPopover() {
            const current =
              selected && selected.tagName.toLowerCase() === "line"
                ? getLineArrowStyle(selected)
                : null;
            arrowPop
              .querySelectorAll("[data-arrow-style]")
              .forEach((btn) =>
                btn.classList.toggle(
                  "is-active",
                  btn.dataset.arrowStyle === current,
                ),
              );
          }

          function snapshot(el) {
            const tag = el.tagName.toLowerCase();
            const num = (k) => Number(el.getAttribute(k)) || 0;
            const base = { angle: getAngle(el) };
            if (tag === "rect") {
              return { ...base, x: num("x"), y: num("y"), w: num("width"), h: num("height") };
            }
            if (tag === "ellipse") {
              return { ...base, cx: num("cx"), cy: num("cy"), rx: num("rx"), ry: num("ry") };
            }
            if (tag === "circle") {
              return { ...base, cx: num("cx"), cy: num("cy"), r: num("r") };
            }
            if (tag === "line") {
              return { ...base, x1: num("x1"), y1: num("y1"), x2: num("x2"), y2: num("y2") };
            }
            if (tag === "text") {
              return { ...base, x: num("x"), y: num("y") };
            }
            if (tag === "polyline" || tag === "polygon") {
              return { ...base, points: el.getAttribute("points") || "" };
            }
            return base;
          }

          function startDraw(pt) {
            let el = null;
            if (tool === "rect") {
              el = document.createElementNS(SVG_NS, "rect");
              el.setAttribute("x", String(pt.x));
              el.setAttribute("y", String(pt.y));
              el.setAttribute("width", "0");
              el.setAttribute("height", "0");
            } else if (tool === "ellipse") {
              el = document.createElementNS(SVG_NS, "ellipse");
              el.setAttribute("cx", String(pt.x));
              el.setAttribute("cy", String(pt.y));
              el.setAttribute("rx", "0");
              el.setAttribute("ry", "0");
            } else if (tool === "line") {
              el = document.createElementNS(SVG_NS, "line");
              el.setAttribute("x1", String(pt.x));
              el.setAttribute("y1", String(pt.y));
              el.setAttribute("x2", String(pt.x));
              el.setAttribute("y2", String(pt.y));
            } else if (tool === "free") {
              el = document.createElementNS(SVG_NS, "polyline");
              el.setAttribute("points", `${pt.x.toFixed(1)},${pt.y.toFixed(1)}`);
            } else if (tool === "text") {
              const text = window.prompt("Text:");
              if (!text) return;
              el = document.createElementNS(SVG_NS, "text");
              el.setAttribute("x", String(pt.x));
              el.setAttribute("y", String(pt.y));
              el.setAttribute("font-family", "sans-serif");
              el.setAttribute("font-size", String(style.fontSize));
              el.textContent = text;
              applyStyles(el);
              contentG.appendChild(el);
              selectEl(el);
              setTool("select");
              commit();
              return;
            }
            if (!el) return;
            applyStyles(el);
            contentG.appendChild(el);
            drag = { mode: "draw", el, startPt: pt, origAttrs: null };
          }

          function updateDraw(pt) {
            if (!drag || !drag.el) return;
            const el = drag.el;
            const s = drag.startPt;
            const tag = el.tagName.toLowerCase();
            if (tag === "rect") {
              el.setAttribute("x", String(Math.min(s.x, pt.x)));
              el.setAttribute("y", String(Math.min(s.y, pt.y)));
              el.setAttribute("width", String(Math.abs(pt.x - s.x)));
              el.setAttribute("height", String(Math.abs(pt.y - s.y)));
            } else if (tag === "ellipse") {
              el.setAttribute("cx", String((s.x + pt.x) / 2));
              el.setAttribute("cy", String((s.y + pt.y) / 2));
              el.setAttribute("rx", String(Math.abs(pt.x - s.x) / 2));
              el.setAttribute("ry", String(Math.abs(pt.y - s.y) / 2));
            } else if (tag === "line") {
              el.setAttribute("x2", String(pt.x));
              el.setAttribute("y2", String(pt.y));
            } else if (tag === "polyline") {
              const cur = el.getAttribute("points") || "";
              const tokens = cur.trim().split(/\s+/);
              const last = tokens[tokens.length - 1] || "0,0";
              const [lx, ly] = last.split(",").map(Number);
              if (Math.hypot(pt.x - lx, pt.y - ly) >= 2) {
                el.setAttribute(
                  "points",
                  `${cur} ${pt.x.toFixed(1)},${pt.y.toFixed(1)}`,
                );
              }
            }
          }

          function finishDraw() {
            if (!drag || !drag.el) {
              drag = null;
              return;
            }
            const el = drag.el;
            const tag = el.tagName.toLowerCase();
            const num = (k) => Number(el.getAttribute(k)) || 0;
            let degenerate = false;
            if (tag === "rect") {
              degenerate = num("width") < 2 || num("height") < 2;
            } else if (tag === "ellipse") {
              degenerate = num("rx") < 1 || num("ry") < 1;
            } else if (tag === "line") {
              degenerate =
                Math.hypot(num("x2") - num("x1"), num("y2") - num("y1")) < 2;
            } else if (tag === "polyline") {
              const pts = (el.getAttribute("points") || "").trim().split(/\s+/);
              degenerate = pts.length < 2;
            }
            if (degenerate) {
              el.remove();
              drag = null;
              return;
            }
            selectEl(el);
            setTool("select");
            drag = null;
            commit();
          }

          function hitTest(target) {
            let n = target;
            while (n && n !== svgEl) {
              if (n.parentNode === contentG) return n;
              n = n.parentNode;
            }
            return null;
          }

          function updateSelectionDependentUi() {
            const has = selection.length > 0;
            const single = selection.length === 1;
            deleteBtn.disabled = !has;
            duplicateBtn.disabled = !single;
            frontBtn.disabled = !single;
            backBtn.disabled = !single;
            const isLine =
              single && selected.tagName.toLowerCase() === "line";
            arrowBtn.disabled = !isLine;
            arrowBtn.classList.toggle(
              "is-active",
              isLine && !!getLineArrowStyle(selected),
            );
            arrowFlyout.classList.toggle("is-enabled", isLine);
            refreshArrowPopover();
          }

          function setSelection(els) {
            selection = (els || []).filter((e) => e && e.parentNode === contentG);
            selected = selection[0] || null;
            updateSelectionDependentUi();
            renderHandles();
            syncStyleControls();
          }

          function selectEl(el) {
            setSelection([el]);
          }

          function selectMany(els) {
            setSelection(els);
          }

          function deselect() {
            setSelection([]);
          }

          function deleteSelected() {
            if (!selection.length) return;
            selection.forEach((el) => el.remove());
            setSelection([]);
            commit();
          }

          function duplicateSelected() {
            if (!selected) return;
            const clone = selected.cloneNode(true);
            const offset = Math.max(6, Math.min(vb.w, vb.h) / 40);
            offsetElement(clone, offset, offset);
            refreshPivot(clone);
            contentG.appendChild(clone);
            selectEl(clone);
            commit();
          }

          function bringForward() {
            if (!selected || !selected.nextElementSibling) return;
            contentG.insertBefore(
              selected.nextElementSibling,
              selected,
            );
            commit();
          }

          function sendBack() {
            if (!selected || !selected.previousElementSibling) return;
            contentG.insertBefore(selected, selected.previousElementSibling);
            commit();
          }

          function offsetElement(el, dx, dy) {
            const tag = el.tagName.toLowerCase();
            const num = (k) => Number(el.getAttribute(k)) || 0;
            if (tag === "rect" || tag === "text") {
              el.setAttribute("x", String(num("x") + dx));
              el.setAttribute("y", String(num("y") + dy));
            } else if (tag === "ellipse" || tag === "circle") {
              el.setAttribute("cx", String(num("cx") + dx));
              el.setAttribute("cy", String(num("cy") + dy));
            } else if (tag === "line") {
              el.setAttribute("x1", String(num("x1") + dx));
              el.setAttribute("y1", String(num("y1") + dy));
              el.setAttribute("x2", String(num("x2") + dx));
              el.setAttribute("y2", String(num("y2") + dy));
            } else if (tag === "polyline" || tag === "polygon") {
              const pts = (el.getAttribute("points") || "")
                .trim()
                .split(/\s+/)
                .map((p) => {
                  const [px, py] = p.split(",").map(Number);
                  return `${(px + dx).toFixed(1)},${(py + dy).toFixed(1)}`;
                })
                .join(" ");
              el.setAttribute("points", pts);
            }
          }

          function normalizeColor(c) {
            if (!c || c === "none") return null;
            if (/^#[0-9a-fA-F]{6}$/.test(c)) return c.toLowerCase();
            if (/^#[0-9a-fA-F]{3}$/.test(c)) {
              return (
                "#" +
                c
                  .slice(1)
                  .split("")
                  .map((x) => x + x)
                  .join("")
                  .toLowerCase()
              );
            }
            const probe = document.createElement("span");
            probe.style.color = c;
            document.body.appendChild(probe);
            const computed = getComputedStyle(probe).color;
            probe.remove();
            const m = computed.match(/rgb\((\d+),\s*(\d+),\s*(\d+)\)/);
            if (!m) return null;
            return (
              "#" +
              [m[1], m[2], m[3]]
                .map((v) => Number(v).toString(16).padStart(2, "0"))
                .join("")
            );
          }

          function syncStyleControls() {
            if (!selected) return;
            const tag = selected.tagName.toLowerCase();
            if (tag === "text") {
              const c = normalizeColor(selected.getAttribute("fill"));
              if (c) {
                style.stroke = c;
                strokeInput.value = c;
              }
              const fs = Number(selected.getAttribute("font-size"));
              if (Number.isFinite(fs) && fs > 0) {
                style.fontSize = fs;
                const nearest = ["10", "12", "14", "16", "20", "24", "32", "48"].reduce(
                  (a, b) =>
                    Math.abs(Number(b) - fs) < Math.abs(Number(a) - fs) ? b : a,
                );
                fontSelect.value = nearest;
              }
              return;
            }
            const sc = normalizeColor(selected.getAttribute("stroke"));
            if (sc) {
              style.stroke = sc;
              strokeInput.value = sc;
            }
            const fillRaw = selected.getAttribute("fill");
            if (!fillRaw || fillRaw === "none") {
              style.fill = "none";
              fillNone.checked = true;
            } else {
              const fc = normalizeColor(fillRaw);
              if (fc) {
                style.fill = fc;
                fillInput.value = fc;
                fillNone.checked = false;
              }
            }
            const sw = Number(selected.getAttribute("stroke-width"));
            if (Number.isFinite(sw) && sw > 0) {
              style.strokeWidth = sw;
              const closest = ["1", "2", "4"].reduce((a, b) =>
                Math.abs(Number(b) - sw) < Math.abs(Number(a) - sw) ? b : a,
              );
              widthSelect.value = closest;
            }
            const da = (selected.getAttribute("stroke-dasharray") || "").trim();
            if (!da) {
              style.dashStyle = "solid";
              dashSelect.value = "solid";
            } else {
              const first = Number(da.split(/[\s,]+/)[0]);
              if (Number.isFinite(first) && first <= style.strokeWidth + 0.5) {
                style.dashStyle = "dotted";
                dashSelect.value = "dotted";
              } else {
                style.dashStyle = "dashed";
                dashSelect.value = "dashed";
              }
            }
          }

          function renderHandles() {
            overlayG.innerHTML = "";
            overlayG.removeAttribute("transform");
            if (selection.length === 0) return;
            if (selection.length > 1) {
              let minX = Infinity,
                minY = Infinity,
                maxX = -Infinity,
                maxY = -Infinity;
              selection.forEach((el) => {
                const b = bboxOf(el);
                if (!b) return;
                minX = Math.min(minX, b.x);
                minY = Math.min(minY, b.y);
                maxX = Math.max(maxX, b.x + b.width);
                maxY = Math.max(maxY, b.y + b.height);
              });
              if (!Number.isFinite(minX)) return;
              const sel = document.createElementNS(SVG_NS, "rect");
              sel.setAttribute("x", String(minX));
              sel.setAttribute("y", String(minY));
              sel.setAttribute("width", String(maxX - minX));
              sel.setAttribute("height", String(maxY - minY));
              sel.setAttribute("class", "mse-selection");
              overlayG.appendChild(sel);
              return;
            }
            const tag = selected.tagName.toLowerCase();
            const handleSize = Math.max(6, Math.min(vb.w, vb.h) / 60);
            const rotateOffset = handleSize * 2.5;
            const addHandle = (x, y, role, cls) => {
              const h = document.createElementNS(SVG_NS, "rect");
              h.setAttribute("x", String(x - handleSize / 2));
              h.setAttribute("y", String(y - handleSize / 2));
              h.setAttribute("width", String(handleSize));
              h.setAttribute("height", String(handleSize));
              h.setAttribute("class", cls || "mse-handle");
              h.dataset.role = role;
              overlayG.appendChild(h);
            };
            const angle = getAngle(selected);
            if (tag === "line" && angle === 0) {
              const x1 = Number(selected.getAttribute("x1"));
              const y1 = Number(selected.getAttribute("y1"));
              const x2 = Number(selected.getAttribute("x2"));
              const y2 = Number(selected.getAttribute("y2"));
              const sel = document.createElementNS(SVG_NS, "line");
              sel.setAttribute("x1", String(x1));
              sel.setAttribute("y1", String(y1));
              sel.setAttribute("x2", String(x2));
              sel.setAttribute("y2", String(y2));
              sel.setAttribute("class", "mse-selection");
              overlayG.appendChild(sel);
              addHandle(x1, y1, "resize-line-p1", "mse-handle mse-handle-move");
              addHandle(x2, y2, "resize-line-p2", "mse-handle mse-handle-move");
              const mx = (x1 + x2) / 2;
              const my = (y1 + y2) / 2;
              addRotationHandle(mx, my - rotateOffset, mx, my);
              return;
            }
            const bbox = bboxOf(selected);
            if (!bbox) return;
            const cx = bbox.x + bbox.width / 2;
            const cy = bbox.y + bbox.height / 2;
            if (angle) {
              overlayG.setAttribute(
                "transform",
                `rotate(${angle} ${cx} ${cy})`,
              );
            }
            const sel = document.createElementNS(SVG_NS, "rect");
            sel.setAttribute("x", String(bbox.x));
            sel.setAttribute("y", String(bbox.y));
            sel.setAttribute("width", String(bbox.width));
            sel.setAttribute("height", String(bbox.height));
            sel.setAttribute("class", "mse-selection");
            overlayG.appendChild(sel);
            if (
              (tag === "rect" || tag === "ellipse" || tag === "circle") &&
              angle === 0
            ) {
              addHandle(
                bbox.x + bbox.width,
                bbox.y + bbox.height,
                "resize-se",
                "mse-handle",
              );
            }
            addRotationHandle(
              cx,
              bbox.y - rotateOffset,
              cx,
              bbox.y,
            );
          }

          function addRotationHandle(hx, hy, anchorX, anchorY) {
            const tether = document.createElementNS(SVG_NS, "line");
            tether.setAttribute("x1", String(anchorX));
            tether.setAttribute("y1", String(anchorY));
            tether.setAttribute("x2", String(hx));
            tether.setAttribute("y2", String(hy));
            tether.setAttribute("class", "mse-rotate-tether");
            overlayG.appendChild(tether);
            const handleSize = Math.max(6, Math.min(vb.w, vb.h) / 60);
            const r = handleSize * 0.6;
            const knob = document.createElementNS(SVG_NS, "circle");
            knob.setAttribute("cx", String(hx));
            knob.setAttribute("cy", String(hy));
            knob.setAttribute("r", String(r));
            knob.setAttribute("class", "mse-handle mse-handle-rotate");
            knob.dataset.role = "rotate";
            overlayG.appendChild(knob);
          }

          function applyMove(el, orig, dx, dy) {
            const tag = el.tagName.toLowerCase();
            if (tag === "rect" || tag === "text") {
              el.setAttribute("x", String(orig.x + dx));
              el.setAttribute("y", String(orig.y + dy));
            } else if (tag === "ellipse" || tag === "circle") {
              el.setAttribute("cx", String(orig.cx + dx));
              el.setAttribute("cy", String(orig.cy + dy));
            } else if (tag === "line") {
              el.setAttribute("x1", String(orig.x1 + dx));
              el.setAttribute("y1", String(orig.y1 + dy));
              el.setAttribute("x2", String(orig.x2 + dx));
              el.setAttribute("y2", String(orig.y2 + dy));
            } else if (tag === "polyline" || tag === "polygon") {
              const pts = orig.points
                .trim()
                .split(/\s+/)
                .map((p) => {
                  const [px, py] = p.split(",").map(Number);
                  return `${(px + dx).toFixed(1)},${(py + dy).toFixed(1)}`;
                })
                .join(" ");
              el.setAttribute("points", pts);
            }
            // Updating the rotation pivot too keeps the rendered shape moving
            // by exactly (dx, dy) in screen space, while preserving the angle
            // and keeping the pivot at the bbox center (see comment in
            // setRotation).
            if (orig.angle) {
              refreshPivot(el);
            }
          }

          function applyResizeSE(el, orig, dx, dy) {
            const tag = el.tagName.toLowerCase();
            if (tag === "rect") {
              el.setAttribute("width", String(Math.max(2, orig.w + dx)));
              el.setAttribute("height", String(Math.max(2, orig.h + dy)));
            } else if (tag === "ellipse") {
              const newRx = Math.max(1, orig.rx + dx / 2);
              const newRy = Math.max(1, orig.ry + dy / 2);
              el.setAttribute("rx", String(newRx));
              el.setAttribute("ry", String(newRy));
              el.setAttribute("cx", String(orig.cx + (newRx - orig.rx)));
              el.setAttribute("cy", String(orig.cy + (newRy - orig.ry)));
            } else if (tag === "circle") {
              const d = Math.max(dx, dy);
              const newR = Math.max(1, orig.r + d / 2);
              el.setAttribute("r", String(newR));
              el.setAttribute("cx", String(orig.cx + (newR - orig.r)));
              el.setAttribute("cy", String(orig.cy + (newR - orig.r)));
            }
          }

          function applyRotation(el, fromPt, toPt, center, origAngle) {
            const a1 = Math.atan2(fromPt.y - center.y, fromPt.x - center.x);
            const a2 = Math.atan2(toPt.y - center.y, toPt.x - center.x);
            let newAngle = origAngle + ((a2 - a1) * 180) / Math.PI;
            newAngle = ((newAngle % 360) + 360) % 360;
            if (newAngle > 180) newAngle -= 360;
            setRotation(el, newAngle, center);
          }

          function onPointerDown(evt) {
            evt.preventDefault();
            const pt = toUserCoords(evt);
            if (spaceDown || evt.button === 1) {
              drag = {
                mode: "pan",
                startScreen: { x: evt.clientX, y: evt.clientY },
                startScroll: {
                  x: canvasWrap.scrollLeft,
                  y: canvasWrap.scrollTop,
                },
              };
              try {
                svgEl.setPointerCapture(evt.pointerId);
              } catch (err) {
                /* pointer capture may be unsupported on legacy browsers */
              }
              canvasWrap.classList.add("is-panning");
              return;
            }
            try {
              svgEl.setPointerCapture(evt.pointerId);
            } catch (err) {
              /* pointer capture may be unsupported on legacy browsers */
            }
            const handle =
              evt.target && evt.target.closest
                ? evt.target.closest(".mse-handle")
                : null;
            if (handle && selection.length === 1) {
              const role = handle.dataset.role;
              if (role === "rotate") {
                const center = bboxCenter(selected);
                drag = {
                  mode: "rotate",
                  el: selected,
                  startPt: pt,
                  origAttrs: snapshot(selected),
                  center: center || { x: pt.x, y: pt.y },
                };
              } else {
                drag = {
                  mode: role,
                  el: selected,
                  startPt: pt,
                  origAttrs: snapshot(selected),
                };
              }
              return;
            }
            if (tool === "select") {
              const hit = hitTest(evt.target);
              if (hit) {
                // Clicking inside the current selection keeps it intact (so
                // you can move the whole group); clicking a different shape
                // narrows to just that shape.
                if (!selection.includes(hit)) selectEl(hit);
                drag = {
                  mode: "move",
                  els: selection.slice(),
                  startPt: pt,
                  origAttrs: selection.map((el) => ({
                    el,
                    attrs: snapshot(el),
                  })),
                };
              } else {
                deselect();
                startMarquee(pt);
              }
            } else {
              startDraw(pt);
            }
          }

          function startMarquee(pt) {
            const m = document.createElementNS(SVG_NS, "rect");
            m.setAttribute("class", "mse-marquee");
            m.setAttribute("x", String(pt.x));
            m.setAttribute("y", String(pt.y));
            m.setAttribute("width", "0");
            m.setAttribute("height", "0");
            overlayG.appendChild(m);
            drag = { mode: "marquee", startPt: pt, marqueeEl: m };
          }

          function updateMarquee(pt) {
            if (!drag || !drag.marqueeEl) return;
            const x = Math.min(drag.startPt.x, pt.x);
            const y = Math.min(drag.startPt.y, pt.y);
            const w = Math.abs(pt.x - drag.startPt.x);
            const h = Math.abs(pt.y - drag.startPt.y);
            drag.marqueeEl.setAttribute("x", String(x));
            drag.marqueeEl.setAttribute("y", String(y));
            drag.marqueeEl.setAttribute("width", String(w));
            drag.marqueeEl.setAttribute("height", String(h));
          }

          function finishMarquee(pt) {
            const x1 = Math.min(drag.startPt.x, pt.x);
            const y1 = Math.min(drag.startPt.y, pt.y);
            const x2 = Math.max(drag.startPt.x, pt.x);
            const y2 = Math.max(drag.startPt.y, pt.y);
            const marqueeArea = (x2 - x1) * (y2 - y1);
            drag.marqueeEl.remove();
            // A marquee that barely moved is treated as a deselect click.
            if (marqueeArea < 4) {
              setSelection([]);
              return;
            }
            const hits = [];
            Array.from(contentG.children).forEach((el) => {
              const b = bboxOf(el);
              if (!b) return;
              if (b.x + b.width < x1 || b.x > x2) return;
              if (b.y + b.height < y1 || b.y > y2) return;
              hits.push(el);
            });
            setSelection(hits);
          }

          function onPointerMove(evt) {
            if (!drag) return;
            if (drag.mode === "pan") {
              canvasWrap.scrollLeft =
                drag.startScroll.x - (evt.clientX - drag.startScreen.x);
              canvasWrap.scrollTop =
                drag.startScroll.y - (evt.clientY - drag.startScreen.y);
              return;
            }
            const pt = toUserCoords(evt);
            const dx = pt.x - drag.startPt.x;
            const dy = pt.y - drag.startPt.y;
            if (drag.mode === "marquee") {
              updateMarquee(pt);
              drag.changed = true;
            } else if (drag.mode === "draw") {
              updateDraw(pt);
              drag.changed = true;
            } else if (drag.mode === "move") {
              drag.origAttrs.forEach(({ el, attrs }) => {
                applyMove(el, attrs, dx, dy);
              });
              renderHandles();
              drag.changed = drag.changed || dx !== 0 || dy !== 0;
            } else if (drag.mode === "resize-se") {
              applyResizeSE(drag.el, drag.origAttrs, dx, dy);
              renderHandles();
              drag.changed = true;
            } else if (drag.mode === "resize-line-p1") {
              drag.el.setAttribute("x1", String(pt.x));
              drag.el.setAttribute("y1", String(pt.y));
              renderHandles();
              drag.changed = true;
            } else if (drag.mode === "resize-line-p2") {
              drag.el.setAttribute("x2", String(pt.x));
              drag.el.setAttribute("y2", String(pt.y));
              renderHandles();
              drag.changed = true;
            } else if (drag.mode === "rotate") {
              applyRotation(
                drag.el,
                drag.startPt,
                pt,
                drag.center,
                drag.origAttrs.angle,
              );
              renderHandles();
              drag.changed = true;
            }
          }

          function onPointerUp(evt) {
            if (!drag) return;
            if (drag.mode === "pan") {
              canvasWrap.classList.remove("is-panning");
              drag = null;
              return;
            }
            if (drag.mode === "marquee") {
              const pt = toUserCoords(evt);
              finishMarquee(pt);
              drag = null;
              return;
            }
            if (drag.mode === "draw") {
              finishDraw();
              return;
            }
            const changed = !!drag.changed;
            drag = null;
            if (changed) commit();
          }

          function onDblClick(evt) {
            const target = evt.target;
            if (!target) return;
            const textEl =
              target.tagName && target.tagName.toLowerCase() === "text"
                ? target
                : null;
            if (textEl && textEl.parentNode === contentG) {
              evt.preventDefault();
              const current = textEl.textContent || "";
              const next = window.prompt("Edit text:", current);
              if (next !== null && next !== current) {
                textEl.textContent = next;
                if (!textEl.textContent) {
                  textEl.remove();
                  deselect();
                } else {
                  refreshPivot(textEl);
                  renderHandles();
                }
                commit();
              }
            }
          }

          function onKey(evt) {
            const t = evt.target;
            const inField =
              t &&
              (t.tagName === "INPUT" ||
                t.tagName === "TEXTAREA" ||
                t.tagName === "SELECT" ||
                t.isContentEditable);
            if (!root.isConnected) return;
            const cmd = evt.ctrlKey || evt.metaKey;
            if (cmd && !evt.shiftKey && evt.key.toLowerCase() === "z") {
              evt.preventDefault();
              undo();
              return;
            }
            if (
              cmd &&
              (evt.key.toLowerCase() === "y" ||
                (evt.key.toLowerCase() === "z" && evt.shiftKey))
            ) {
              evt.preventDefault();
              redo();
              return;
            }
            if (cmd && evt.key.toLowerCase() === "d" && selected) {
              evt.preventDefault();
              duplicateSelected();
              return;
            }
            if (inField) return;
            if (evt.code === "Space") {
              if (!spaceDown) {
                spaceDown = true;
                canvasWrap.classList.add("is-pan-ready");
              }
              evt.preventDefault();
              return;
            }
            if (
              (evt.key === "Delete" || evt.key === "Backspace") &&
              selection.length
            ) {
              evt.preventDefault();
              deleteSelected();
              return;
            }
            if (evt.key === "Escape") {
              if (drag && drag.mode === "draw" && drag.el) {
                drag.el.remove();
              }
              if (drag && drag.mode === "marquee" && drag.marqueeEl) {
                drag.marqueeEl.remove();
              }
              drag = null;
              deselect();
              return;
            }
            if (
              selection.length &&
              (evt.key === "ArrowUp" ||
                evt.key === "ArrowDown" ||
                evt.key === "ArrowLeft" ||
                evt.key === "ArrowRight")
            ) {
              evt.preventDefault();
              const step = evt.shiftKey ? 10 : 1;
              const dx =
                evt.key === "ArrowRight"
                  ? step
                  : evt.key === "ArrowLeft"
                    ? -step
                    : 0;
              const dy =
                evt.key === "ArrowDown"
                  ? step
                  : evt.key === "ArrowUp"
                    ? -step
                    : 0;
              selection.forEach((el) => {
                offsetElement(el, dx, dy);
                refreshPivot(el);
              });
              renderHandles();
              commit();
            }
          }

          function onKeyUp(evt) {
            if (evt.code === "Space") {
              spaceDown = false;
              canvasWrap.classList.remove("is-pan-ready");
            }
          }

          function onWheel(evt) {
            if (!(evt.ctrlKey || evt.metaKey)) return;
            evt.preventDefault();
            const factor = evt.deltaY > 0 ? 1 / 1.1 : 1.1;
            setZoom(zoom * factor);
          }

          svgEl.addEventListener("pointerdown", onPointerDown);
          svgEl.addEventListener("pointermove", onPointerMove);
          svgEl.addEventListener("pointerup", onPointerUp);
          svgEl.addEventListener("pointercancel", onPointerUp);
          svgEl.addEventListener("dblclick", onDblClick);
          canvasWrap.addEventListener("wheel", onWheel, { passive: false });
          document.addEventListener("keydown", onKey);
          document.addEventListener("keyup", onKeyUp);

          function snapshotState() {
            return {
              defs: defsEl.innerHTML,
              content: contentG.innerHTML,
              vb: { ...vb },
            };
          }

          function restoreState(state) {
            defsEl.innerHTML = state.defs;
            contentG.innerHTML = state.content;
            vb = { ...state.vb };
            applyViewBoxAttrs();
            deselect();
          }

          function commit() {
            if (historyIdx < history.length - 1) {
              history.splice(historyIdx + 1);
            }
            history.push(snapshotState());
            if (history.length > HISTORY_LIMIT) {
              history.shift();
            } else {
              historyIdx++;
            }
            updateHistoryUi();
          }

          function updateHistoryUi() {
            undoBtn.disabled = historyIdx <= 0;
            redoBtn.disabled = historyIdx >= history.length - 1;
          }

          function undo() {
            if (historyIdx <= 0) return;
            historyIdx--;
            restoreState(history[historyIdx]);
            updateHistoryUi();
          }

          function redo() {
            if (historyIdx >= history.length - 1) return;
            historyIdx++;
            restoreState(history[historyIdx]);
            updateHistoryUi();
          }

          function setSVG(svgString) {
            deselect();
            drag = null;
            defsEl.innerHTML = "";
            contentG.innerHTML = "";
            if (!svgString || !String(svgString).trim()) {
              vb = { x: 0, y: 0, w: defaultWidth, h: defaultHeight };
              applyViewBoxAttrs();
              history.length = 0;
              historyIdx = -1;
              commit();
              return true;
            }
            try {
              // Parse via an HTML <template>, not DOMParser. DOMParser with
              // "image/svg+xml" leaves children in a *null* namespace if the
              // source <svg> didn't carry an xmlns attribute -- which is
              // common because HTML serialisers don't emit it. Null-namespace
              // children render as inert HTML elements in our SVG canvas, so
              // the canvas looks empty even though the DOM is populated.
              // The template's HTML parser correctly infers the SVG
              // namespace from the <svg> tag.
              const tmpl = document.createElement("template");
              tmpl.innerHTML = String(svgString);
              const incoming = tmpl.content.querySelector("svg");
              if (!incoming) return false;
              const vbAttr = incoming.getAttribute("viewBox");
              if (vbAttr) {
                const p = vbAttr.split(/[\s,]+/).map(Number);
                if (p.length === 4 && p.every((n) => Number.isFinite(n))) {
                  vb = { x: p[0], y: p[1], w: p[2], h: p[3] };
                }
              } else {
                const w = Number(incoming.getAttribute("width")) || defaultWidth;
                const h = Number(incoming.getAttribute("height")) || defaultHeight;
                vb = { x: 0, y: 0, w, h };
              }
              applyViewBoxAttrs();
              Array.from(incoming.childNodes).forEach((node) => {
                if (node.nodeType !== 1) return;
                const tag = node.tagName.toLowerCase();
                if (tag === "script") return;
                const adopted = document.importNode(node, true);
                if (tag === "defs") {
                  Array.from(adopted.childNodes).forEach((d) => {
                    defsEl.appendChild(d.cloneNode(true));
                  });
                } else {
                  contentG.appendChild(adopted);
                }
              });
              // For shapes that came in with a rotate transform but no
              // data-mse-angle (e.g. hand-written or older exports),
              // recover the angle so the rotation handle can drive it
              // additively rather than resetting on first drag.
              contentG.querySelectorAll("[transform]").forEach((el) => {
                if (el.dataset.mseAngle != null) return;
                const t = el.getAttribute("transform") || "";
                const m = t.match(/rotate\(\s*(-?[\d.]+)/);
                if (m) el.dataset.mseAngle = m[1];
              });
              history.length = 0;
              historyIdx = -1;
              commit();
              return true;
            } catch (err) {
              console.error("Mini SVG editor setSVG failed:", err);
              return false;
            }
          }

          function getSVG() {
            const clone = svgEl.cloneNode(true);
            const ov = clone.querySelector(".mse-overlay");
            if (ov) ov.remove();
            const cg = clone.querySelector(".mse-content");
            if (cg) {
              const parent = cg.parentNode;
              while (cg.firstChild) {
                parent.insertBefore(cg.firstChild, cg);
              }
              cg.remove();
            }
            const cloneDefs = clone.querySelector("defs");
            if (cloneDefs && !cloneDefs.children.length) {
              cloneDefs.remove();
            }
            clone.removeAttribute("class");
            // Strip our zoom display width/height so the saved SVG renders at
            // intrinsic size (viewBox already encodes the aspect ratio).
            // data-mse-angle is preserved so reopening the editor restores
            // the rotation state.
            clone.setAttribute("width", String(vb.w));
            clone.setAttribute("height", String(vb.h));
            if (!clone.getAttribute("xmlns")) {
              clone.setAttribute("xmlns", SVG_NS);
            }
            return clone.outerHTML;
          }

          function destroy() {
            svgEl.removeEventListener("pointerdown", onPointerDown);
            svgEl.removeEventListener("pointermove", onPointerMove);
            svgEl.removeEventListener("pointerup", onPointerUp);
            svgEl.removeEventListener("pointercancel", onPointerUp);
            svgEl.removeEventListener("dblclick", onDblClick);
            canvasWrap.removeEventListener("wheel", onWheel);
            document.removeEventListener("keydown", onKey);
            document.removeEventListener("keyup", onKeyUp);
            if (root.parentNode) root.parentNode.removeChild(root);
          }

          updateHistoryUi();
          updateSelectionDependentUi();

          return { setSVG, getSVG, destroy };
        }

        let miniEditor = null;

        function loadSvgEditor() {
          if (miniEditor) return Promise.resolve(miniEditor);
          const dialog = ensureDiagramEditorDialog();
          const host = dialog.querySelector("#svgedit-host");
          host.innerHTML = "";
          miniEditor = createMiniSvgEditor(host, { width: 640, height: 480 });
          state.svgEditor = miniEditor;
          return Promise.resolve(miniEditor);
        }

        function svgEditorSetSvg(svgString) {
          return miniEditor ? miniEditor.setSVG(svgString) : false;
        }

        function svgEditorGetSvg() {
          return miniEditor ? miniEditor.getSVG() : null;
        }

        async function prepareSvgEditor(dialog, svgString) {
          const statusEl = dialog.querySelector("#svgedit-status");
          state.svgEditReady = false;
          statusEl.hidden = false;
          statusEl.textContent = "Loading visual editor…";

          try {
            await loadSvgEditor();
            if (!state.diagramEditContext) return false;
            // Yield one frame so the dialog (just opened a microtask ago)
            // has had a layout pass before the SVG content is mounted.
            // Without this the first-open render can show an empty canvas
            // even though the children are in the DOM -- the painter only
            // catches up once the user toggles tabs.
            await new Promise((r) => requestAnimationFrame(r));
            if (!state.diagramEditContext) return false;
            svgEditorSetSvg(svgString);
            state.svgEditReady = true;
            statusEl.hidden = true;
            return true;
          } catch (err) {
            console.error("SVG editor failed to load:", err);
            statusEl.textContent =
              "Visual editor unavailable (offline or blocked) — use the SVG source tab instead.";
            if (state.diagramEditContext && state.diagramEditTab === "visual") {
              setDiagramTabUi(dialog, "source");
              renderDiagramSourcePreview(dialog);
            }
            return false;
          }
        }

        function finishDiagramEdit(dialog, ctx) {
          let svgString = null;
          if (state.diagramEditTab === "visual" && state.svgEditReady) {
            svgString = svgEditorGetSvg();
          }
          if (typeof svgString !== "string" || !svgString.trim()) {
            svgString = dialog.querySelector("#diagram-svg-source").value;
          }
          const clean = sanitizeSvgString(svgString);
          if (!clean) {
            showMessage({
              title: "Diagram not saved",
              message: "The edited content is not a valid SVG element.",
              confirmLabel: "OK",
            });
            return;
          }
          const widthValue = dialog
            .querySelector("#diagram-display-width")
            .value.trim();
          const final = applyDiagramDisplayWidth(clean, widthValue);
          if (ctx.onConfirm) {
            ctx.onConfirm(final);
          }
        }

        function applyDiagramDisplayWidth(svgString, widthValue) {
          const tmp = document.createElement("div");
          tmp.innerHTML = svgString;
          const svg = tmp.querySelector("svg");
          if (!svg) return svgString;
          const width = Number(widthValue);
          if (widthValue && Number.isFinite(width) && width > 0) {
            svg.style.width = `${clamp(Math.round(width), 40, 680)}px`;
            svg.style.height = "auto";
          } else {
            svg.style.removeProperty("width");
            svg.style.removeProperty("height");
            if (!svg.getAttribute("style")) {
              svg.removeAttribute("style");
            }
          }
          return svg.outerHTML;
        }

        function readDiagramDisplayWidth(svgString) {
          const tmp = document.createElement("div");
          tmp.innerHTML = svgString;
          const svg = tmp.querySelector("svg");
          if (!svg) return "";
          const m = (svg.getAttribute("style") || "").match(
            /width:\s*([\d.]+)px/,
          );
          return m ? String(Math.round(Number(m[1]))) : "";
        }

        function openDiagramEditorDialog(svgString, onConfirm) {
          const dialog = ensureDiagramEditorDialog();
          const sourceArea = dialog.querySelector("#diagram-svg-source");
          const initial = svgString || STARTER_SVG;
          sourceArea.value = formatHtmlSource(initial);
          dialog.querySelector("#diagram-display-width").value =
            readDiagramDisplayWidth(initial);
          state.diagramEditContext = { onConfirm };
          dialog.returnValue = "";
          setDiagramTabUi(dialog, "visual");
          renderDiagramSourcePreview(dialog);
          showAsNonModal(dialog);
          prepareSvgEditor(dialog, initial).catch((err) =>
            console.error("SVG editor failed to start:", err),
          );
        }

        /* ----------- In-place embed editing from the worksheet ------------ */

        function embedKind(el) {
          if (el.classList && el.classList.contains("fn-graph")) {
            return "fn-graph";
          }
          if (el.tagName && el.tagName.toLowerCase() === "img") return "img";
          return "svg";
        }

        function updateCachedEmbedSource(embedEl, kind, mutate) {
          const target = embedEl.closest("[data-original-latex]");
          if (!target) return;
          const selector =
            kind === "fn-graph" ? ".fn-graph" : kind === "img" ? "img" : "svg";
          const isCandidate = (el) =>
            kind === "fn-graph" ||
            !el.closest(".fn-graph, .katex, .marks, .no-pdf");

          const live = Array.from(target.querySelectorAll(selector)).filter(
            isCandidate,
          );
          const index = live.indexOf(embedEl);
          if (index < 0) return;

          const doc = new DOMParser().parseFromString(
            `<div id="__root">${target.dataset.originalLatex || ""}</div>`,
            "text/html",
          );
          const root = doc.getElementById("__root");
          if (!root) return;
          const cached = Array.from(root.querySelectorAll(selector)).filter(
            isCandidate,
          );
          const cachedEl = cached[index];
          if (!cachedEl) return;
          mutate(cachedEl, doc);
          target.dataset.originalLatex = root.innerHTML;
        }

        function applyGraphEditInPlace(div, spec) {
          const json = JSON.stringify(spec);
          updateCachedEmbedSource(div, "fn-graph", (cachedEl) => {
            cachedEl.setAttribute("data-graph", json);
            cachedEl.innerHTML = "";
          });
          div.setAttribute("data-graph", json);
          renderFunctionGraph(div);
          scheduleLayout();
        }

        function applyDiagramEditInPlace(svg, newSvgString) {
          updateCachedEmbedSource(svg, "svg", (cachedEl, doc) => {
            const tmp = doc.createElement("div");
            tmp.innerHTML = newSvgString;
            const next = tmp.querySelector("svg");
            if (next) {
              next.classList.remove("worksheet-diagram");
              if (!next.getAttribute("class")) {
                next.removeAttribute("class");
              }
              cachedEl.replaceWith(next);
            }
          });
          const tmp = document.createElement("div");
          tmp.innerHTML = newSvgString;
          const next = tmp.querySelector("svg");
          if (next) {
            svg.replaceWith(next);
            const scope = next.closest(".question-prompt, .answer-body");
            decorateDiagramSvgs(scope || els.content);
            decorateEmbedMarkers();
          }
          scheduleLayout();
        }

        function applyImageEditInPlace(img, dataUrl) {
          if (!dataUrl) return;
          updateCachedEmbedSource(img, "img", (cachedEl) => {
            cachedEl.setAttribute("src", dataUrl);
          });
          img.setAttribute("src", dataUrl);
          decorateEmbedMarkers();
          scheduleLayout();
        }

        function openEmbedEditor(embed) {
          const kind = embedKind(embed);
          if (kind === "fn-graph") {
            const spec =
              normalizeGraphSpec(embed.getAttribute("data-graph")) ||
              cloneDefaultGraphSpec();
            openGraphEditorDialog(spec, (newSpec) =>
              applyGraphEditInPlace(embed, newSpec),
            );
          } else if (kind === "img") {
            openImageEditor(embed.getAttribute("src"), (dataUrl) =>
              applyImageEditInPlace(embed, dataUrl),
            );
          } else {
            openDiagramEditorDialog(embed.outerHTML, (newSvg) =>
              applyDiagramEditInPlace(embed, newSvg),
            );
          }
        }

        async function deleteEmbed(embed) {
          const kind = embedKind(embed);
          const label =
            kind === "fn-graph"
              ? "graph"
              : kind === "img"
                ? "image"
                : "diagram";
          const ok = await showMessage({
            title: `Delete ${label}`,
            message: `Delete this ${label} from the worksheet?`,
            confirmLabel: "Delete",
            cancelLabel: "Cancel",
            danger: true,
          });
          if (!ok) return;
          updateCachedEmbedSource(embed, kind, (cachedEl) => cachedEl.remove());
          embed.remove();
          decorateEmbedMarkers();
          scheduleLayout();
        }

        function resizeEmbed(embed, factor) {
          const kind = embedKind(embed);
          if (kind === "fn-graph") {
            const spec = normalizeGraphSpec(embed.getAttribute("data-graph"));
            if (!spec) return;
            spec.width = clamp(Math.round(spec.width * factor), 120, 680);
            spec.height = clamp(Math.round(spec.height * factor), 100, 600);
            applyGraphEditInPlace(embed, spec);
          } else if (kind === "img") {
            const current = embed.getBoundingClientRect().width || 100;
            const next = clamp(Math.round(current * factor), 24, 680);
            updateCachedEmbedSource(embed, "img", (cachedEl) => {
              cachedEl.removeAttribute("width");
              cachedEl.removeAttribute("height");
              cachedEl.style.width = `${next}px`;
              cachedEl.style.height = "auto";
            });
            embed.removeAttribute("width");
            embed.removeAttribute("height");
            embed.style.width = `${next}px`;
            embed.style.height = "auto";
            decorateEmbedMarkers();
            scheduleLayout();
          } else {
            const current = embed.getBoundingClientRect().width || 200;
            const next = clamp(Math.round(current * factor), 40, 680);
            updateCachedEmbedSource(embed, "svg", (cachedEl) => {
              cachedEl.style.width = `${next}px`;
              cachedEl.style.height = "auto";
            });
            embed.style.width = `${next}px`;
            embed.style.height = "auto";
            scheduleLayout();
          }
        }

        /* ------------------------- Embed markers --------------------------
         * Each graph/diagram gets the same right-margin expanding marker as
         * questions and answers. Markers are appended to the host block
         * (like .pagination-tools, stripped on save) and vertically aligned
         * with the embed during the layout pass.
         * ------------------------------------------------------------------ */

        function decorateEmbedMarkers() {
          els.content
            .querySelectorAll(".embed-tools")
            .forEach((marker) => marker.remove());

          // SVG diagrams and function graphs are edited via the question
          // editor's double-click flow, so the in-margin embed panel is only
          // useful for images now.
          els.content
            .querySelectorAll("img")
            .forEach((embed) => {
              const block = embed.closest(".question, .answer");
              if (!block) return;
              if (
                !embed.closest(".question-prompt, .answer-body") ||
                embed.closest(".no-pdf")
              ) {
                return;
              }
              const kind = embedKind(embed);
              const noun =
                kind === "fn-graph"
                  ? "graph"
                  : kind === "img"
                    ? "image"
                    : "diagram";
              const editLabel = `Edit ${noun}`;
              const deleteLabel = `Delete ${noun}`;
              const marker = document.createElement("span");
              marker.className = "content-element-tools embed-tools no-pdf";
              marker.dataset.rowCount = "1";
              marker.innerHTML = `
                <span class="tool-rows-stack">
                  <span class="tool-row" aria-label="${editLabel} controls">
                    <button class="edit-button" type="button" data-action="embed-edit" aria-label="${editLabel}" title="${editLabel}">
                      <svg width="15" height="15" viewBox="0 0 24 24" aria-hidden="true">
                        <path d="M4 20h4l10.5-10.5-4-4L4 16v4z" fill="none" stroke="currentColor" stroke-width="1.8" stroke-linejoin="round"></path>
                        <path d="m13.5 6.5 4 4" fill="none" stroke="currentColor" stroke-width="1.8" stroke-linecap="round"></path>
                      </svg>
                    </button>
                    <button class="delete-button" type="button" data-action="embed-delete" aria-label="${deleteLabel}" title="${deleteLabel}">
                      <svg width="15" height="15" viewBox="0 0 24 24" aria-hidden="true">
                        <path d="M5 7h14M9 7V5h6v2M7 7l1 13h8l1-13" fill="none" stroke="currentColor" stroke-width="1.8" stroke-linecap="round" stroke-linejoin="round"></path>
                      </svg>
                    </button>
                    <span class="action-divider"></span>
                    <button type="button" data-action="embed-shrink" aria-label="Reduce display size" title="Reduce display size">−</button>
                    <button type="button" data-action="embed-grow" aria-label="Increase display size" title="Increase display size">+</button>
                  </span>
                </span>
                ${hoverTriggerHtml(`Show ${noun} controls`)}
              `;
              marker._embedTarget = embed;
              block.appendChild(marker);
              wireToolbarHover(marker);
            });

          positionEmbedMarkers();
        }

        function positionEmbedMarkers() {
          els.content.querySelectorAll(".embed-tools").forEach((marker) => {
            const embed = marker._embedTarget;
            const block = marker.parentElement;
            if (!embed || !embed.isConnected || !block) {
              marker.remove();
              return;
            }
            const blockRect = block.getBoundingClientRect();
            const embedRect = embed.getBoundingClientRect();
            marker.style.top = `${Math.round(
              embedRect.top - blockRect.top,
            )}px`;
          });
        }

        function ensureLatexEditDialog() {
          let dialog = document.getElementById("latex-edit-dialog");
          if (dialog) return dialog;
          dialog = document.createElement("dialog");
          dialog.id = "latex-edit-dialog";
          dialog.className = "no-pdf";
          dialog.innerHTML = `
            <form method="dialog" class="edit-dialog-form">
              <h3>Edit equation</h3>
              <div class="latex-edit-modes" role="group" aria-label="Editor mode">
                <button type="button" data-mode="latex" aria-pressed="true">LaTeX source</button>
                <button type="button" data-mode="visual" aria-pressed="false">Equation builder</button>
              </div>
              <div class="latex-edit-pane" data-pane="latex">
                <textarea id="latex-edit-source" spellcheck="false"></textarea>
              </div>
              <div class="latex-edit-pane" data-pane="visual" hidden>
                <div class="latex-edit-mathfield-wrap">
                  <math-field id="latex-edit-mathfield"></math-field>
                </div>
              </div>
              <label class="latex-edit-display-toggle">
                <input type="checkbox" id="latex-edit-display"> Display (block) equation
              </label>
              <div class="latex-edit-preview-label">Preview</div>
              <div id="latex-edit-preview"></div>
              <div class="edit-dialog-actions">
                <button type="button" data-action="latex-cancel">Cancel</button>
                <button type="submit" value="ok" data-action="latex-ok">OK</button>
              </div>
            </form>
          `;
          document.body.appendChild(dialog);

          const sourceArea = dialog.querySelector("#latex-edit-source");
          const mathfield = dialog.querySelector("#latex-edit-mathfield");
          const displayToggle = dialog.querySelector("#latex-edit-display");
          const preview = dialog.querySelector("#latex-edit-preview");

          const renderPreview = () => {
            const latex = sourceArea.value;
            const block = displayToggle.checked;
            preview.innerHTML = "";
            if (window.katex) {
              try {
                window.katex.render(latex, preview, {
                  throwOnError: false,
                  displayMode: block,
                  output: "html",
                });
                return;
              } catch {}
            }
            preview.textContent = latex;
          };

          sourceArea.addEventListener("input", () => {
            if (mathfield && state.latexEditMode !== "visual") {
              try {
                mathfield.value = sourceArea.value;
              } catch {}
            }
            renderPreview();
          });
          displayToggle.addEventListener("change", renderPreview);

          if (mathfield) {
            mathfield.addEventListener("input", () => {
              if (state.latexEditMode === "visual") {
                sourceArea.value = mathfield.value || "";
                renderPreview();
              }
            });
          }

          dialog.querySelectorAll(".latex-edit-modes button").forEach((btn) => {
            btn.addEventListener("click", () => {
              const mode = btn.dataset.mode;
              state.latexEditMode = mode;
              dialog
                .querySelectorAll(".latex-edit-modes button")
                .forEach((b) =>
                  b.setAttribute(
                    "aria-pressed",
                    String(b.dataset.mode === mode),
                  ),
                );
              dialog.querySelectorAll(".latex-edit-pane").forEach((pane) => {
                pane.hidden = pane.dataset.pane !== mode;
              });
              if (mode === "visual" && mathfield) {
                try {
                  mathfield.value = sourceArea.value;
                } catch {}
              } else if (mode === "latex" && mathfield) {
                sourceArea.value = mathfield.value || sourceArea.value;
                renderPreview();
              }
            });
          });

          dialog
            .querySelector('[data-action="latex-cancel"]')
            .addEventListener("click", () => dialog.close("cancel"));

          dialog.addEventListener("close", () => {
            if (window.mathVirtualKeyboard) {
              try {
                window.mathVirtualKeyboard.visible = false;
                window.mathVirtualKeyboard.container = document.body;
              } catch {}
            }
            if (dialog.returnValue !== "ok") {
              state.latexEditContext = null;
              return;
            }
            const latex = sourceArea.value.trim();
            const display = displayToggle.checked;
            const ctx = state.latexEditContext;
            state.latexEditContext = null;
            if (!latex) return;
            if (ctx && ctx.onConfirm) {
              ctx.onConfirm(latex, display);
            } else if (ctx && ctx.node) {
              ctx.node.setAttribute("data-latex", latex);
              renderLatexNode(ctx.node, latex, display);

        const nonModalStack = [];

        function showAsNonModal(dialog) {
          if (dialog.open) return;

          const idx = nonModalStack.length;
          const baseZ = 50 + idx * 10;
          const dialogZ = baseZ + 1;

          dialog.style.zIndex = String(dialogZ);
          dialog.classList.add("is-nonmodal");

          let backdrop = document.getElementById("dialog-backdrop");
          if (!backdrop) {
            backdrop = document.createElement("div");
            backdrop.id = "dialog-backdrop";
            backdrop.className = "dialog-backdrop no-pdf";
            document.body.appendChild(backdrop);
          }
          backdrop.style.zIndex = String(baseZ);
          backdrop.classList.add("is-visible");
          backdrop.onclick = () => {
            const top = nonModalStack[nonModalStack.length - 1];
            if (top && top.open) top.close("cancel");
          };

          const onEsc = (event) => {
            if (
              event.key === "Escape" &&
              nonModalStack[nonModalStack.length - 1] === dialog &&
              dialog.open
            ) {
              event.preventDefault();
              dialog.close("cancel");
            }
          };
          document.addEventListener("keydown", onEsc);

          const cleanup = () => {
            const i = nonModalStack.indexOf(dialog);
            if (i >= 0) nonModalStack.splice(i, 1);
            dialog.classList.remove("is-nonmodal");
            dialog.style.zIndex = "";
            document.removeEventListener("keydown", onEsc);
            dialog.removeEventListener("close", cleanup);

            if (nonModalStack.length === 0) {
              backdrop.classList.remove("is-visible");
              backdrop.style.zIndex = "";
              backdrop.onclick = null;
            } else {
              const top = nonModalStack[nonModalStack.length - 1];
              const topZ = Number(top.style.zIndex) || 51;
              backdrop.style.zIndex = String(topZ - 1);
            }
          };
          dialog.addEventListener("close", cleanup);

          nonModalStack.push(dialog);
          dialog.show();
        }

        function downsampleToDataUrl(source, srcW, srcH, maxDim, quality) {
          const ratio = Math.min(maxDim / srcW, maxDim / srcH, 1);
          const w = Math.max(1, Math.round(srcW * ratio));
          const h = Math.max(1, Math.round(srcH * ratio));
          const canvas = document.createElement("canvas");
          canvas.width = w;
          canvas.height = h;
          const ctx2 = canvas.getContext("2d");
          ctx2.imageSmoothingEnabled = true;
          ctx2.imageSmoothingQuality = "high";
          ctx2.drawImage(source, 0, 0, w, h);
          return canvas.toDataURL("image/jpeg", quality);
        }

        function approxBase64Bytes(dataUrl) {
          const i = dataUrl.indexOf(",");
          const data = i >= 0 ? dataUrl.slice(i + 1) : dataUrl;
          return Math.round((data.length * 3) / 4);
        }

        function formatBytes(bytes) {
          if (bytes < 1024) return `${bytes} B`;
          if (bytes < 1024 * 1024) return `${(bytes / 1024).toFixed(1)} KB`;
          return `${(bytes / 1024 / 1024).toFixed(2)} MB`;
        }

        function readMarksValue(marks) {
          if (marks.dataset.marksValue !== undefined) {
            const stored = Number(marks.dataset.marksValue);
            if (Number.isFinite(stored) && stored >= 1) {
              return Math.round(stored);
            }
          }
          const match = (marks.textContent || "").match(/(\d+)/);
          return match ? Math.max(1, parseInt(match[1], 10)) : 1;
        }

        function formatMarksText(value) {
          const v = Math.max(1, Math.round(value));
          return `${v} ${v === 1 ? "mark" : "marks"}`;
        }

        function adjustMarks(marks, delta) {
          const current = readMarksValue(marks);
          const next = Math.max(1, current + delta);
          if (next === current) {
            return;
          }
          marks.dataset.marksValue = String(next);
          const text = marks.querySelector(":scope > .marks-text");
          if (text) {
            text.textContent = formatMarksText(next);
          }
          scheduleLayout();
        }

        function moveQuestion(block, direction) {
          const questions = Array.from(
            els.content.querySelectorAll(".question"),
          );
          const idx = questions.indexOf(block);
          if (idx < 0) {
            return;
          }

          const targetIdx = idx + direction;
          const target = questions[targetIdx];
          if (!target) {
            return;
          }

          const answers = Array.from(
            els.content.querySelectorAll(".answer"),
          );
          const blockAnswer = answers[idx];
          const targetAnswer = answers[targetIdx];

          if (direction < 0) {
            target.parentNode.insertBefore(block, target);
            if (blockAnswer && targetAnswer) {
              targetAnswer.parentNode.insertBefore(blockAnswer, targetAnswer);
            }
          } else {
            target.parentNode.insertBefore(block, target.nextSibling);
            if (blockAnswer && targetAnswer) {
              targetAnswer.parentNode.insertBefore(
                blockAnswer,
                targetAnswer.nextSibling,
              );
            }
          }

          scheduleLayout();
        }

          if (hasSpacing) {
            // Questions get two separate toolbars:
            // - header tools (edit/move/delete) anchored at the top-right of the question
            // - spacing tools (resize/break) anchored at the end of the working space
            const headerTools = document.createElement("div");
            headerTools.className =
              "pagination-tools pagination-tools--header no-pdf";
            headerTools.dataset.rowCount = "1";
            headerTools.innerHTML = `
              <div class="tool-rows-stack">${questionActionsRow}</div>
              ${hoverTriggerHtml("Show question controls")}
            `;
            block.appendChild(headerTools);
            wireToolbarHover(headerTools);

            const spacer = document.createElement("div");
            spacer.className = "manual-break-spacer no-pdf";
            spacer.setAttribute("aria-hidden", "true");
            block.appendChild(spacer);

            const spacingTools = document.createElement("div");
            spacingTools.className =
              "pagination-tools pagination-tools--spacing no-pdf";
            spacingTools.dataset.rowCount = "1";
            spacingTools.innerHTML = `
              <div class="tool-rows-stack">${spacingRow}</div>
              ${hoverTriggerHtml("Show spacing controls")}
            `;
            block.appendChild(spacingTools);
            wireToolbarHover(spacingTools);
          } else {
            const spacer = document.createElement("div");
            spacer.className = "manual-break-spacer no-pdf";
            spacer.setAttribute("aria-hidden", "true");
            block.appendChild(spacer);

            const tools = document.createElement("div");
            tools.className = "pagination-tools no-pdf";
            tools.dataset.rowCount = "1";
            tools.innerHTML = `
              <div class="tool-rows-stack">${answerRow}</div>
              ${hoverTriggerHtml("Show answer controls")}
            `;
            block.appendChild(tools);
            wireToolbarHover(tools);
          }
        }

        function hoverTriggerHtml(label) {
          return `
            <button class="hover-trigger" type="button" aria-label="${label}" aria-expanded="false" title="${label}">
              <svg width="10" height="14" viewBox="0 0 10 14" aria-hidden="true">
                <circle cx="5" cy="2.5" r="1.2" fill="currentColor"></circle>
                <circle cx="5" cy="7" r="1.2" fill="currentColor"></circle>
                <circle cx="5" cy="11.5" r="1.2" fill="currentColor"></circle>
              </svg>
            </button>
          `;
        }

        function wireToolbarHover(container) {
          if (container.dataset.hoverWired === "true") {
            return;
          }
          container.dataset.hoverWired = "true";

          let hideTimer = 0;
          const cancelHide = () => {
            if (hideTimer) {
              clearTimeout(hideTimer);
              hideTimer = 0;
            }
          };
          const open = () => {
            cancelHide();
            container.classList.add("is-open");
            const trigger = container.querySelector(":scope > .hover-trigger");
            if (trigger) {
              trigger.setAttribute("aria-expanded", "true");
            }
          };
          const scheduleClose = () => {
            cancelHide();
            hideTimer = setTimeout(() => {
              container.classList.remove("is-open");
              const trigger = container.querySelector(
                ":scope > .hover-trigger",
              );
              if (trigger) {
                trigger.setAttribute("aria-expanded", "false");
              }
              hideTimer = 0;
            }, 1000);
          };

          const trigger = container.querySelector(":scope > .hover-trigger");
          const stack = container.querySelector(":scope > .tool-rows-stack");
          [trigger, stack].forEach((el) => {
            if (!el) {
              return;
            }
            el.addEventListener("pointerenter", open);
            el.addEventListener("pointerleave", scheduleClose);
            el.addEventListener("focusin", open);
            el.addEventListener("focusout", scheduleClose);
          });
        }

        function wireControls() {
          els.content.addEventListener("pointerdown", (event) => {
            const handle = event.target.closest('[data-action="space-drag"]');
            if (!handle) {
              return;
            }

            startSpaceDrag(event, handle);
          });

          els.content.addEventListener("dblclick", (event) => {
            const embed = event.target.closest(
              ".fn-graph, svg.worksheet-diagram, .question-prompt img, .answer-body img",
            );
            if (!embed || embed.closest(".no-pdf")) {
              return;
            }
            event.preventDefault();
            openEmbedEditor(embed);
          });

          els.content.addEventListener("click", (event) => {
            const button = event.target.closest("button[data-action]");
            if (!button) {
              return;
            }

            const action = button.dataset.action;

            if (action === "edit-content") {
              const target = button.closest(".editable-text-target");
              openEditDialog(
                target,
                button.dataset.editLabel || "Edit content",
              );
              return;
            }

            if (action === "delete-content") {
              const target = button.closest(".editable-text-target");
              deleteStandaloneEditableText(
                target,
                button.dataset.deleteLabel || "Delete worksheet element",
              );
              return;
            }

            if (
              action === "embed-edit" ||
              action === "embed-delete" ||
              action === "embed-shrink" ||
              action === "embed-grow"
            ) {
              const marker = button.closest(".embed-tools");
              const embed = marker && marker._embedTarget;
              if (!embed || !embed.isConnected) {
                return;
              }
              if (action === "embed-edit") {
                openEmbedEditor(embed);
              } else if (action === "embed-delete") {
                deleteEmbed(embed);
              } else {
                resizeEmbed(embed, action === "embed-grow" ? 1.1 : 1 / 1.1);
              }
              return;
            }

            if (
              action === "marks-up" ||
              action === "marks-down" ||
              action === "marks-toggle"
            ) {
              const marks = button.closest(".marks");
              if (!marks) {
                return;
              }
              if (action === "marks-up") {
                adjustMarks(marks, 1);
              } else if (action === "marks-down") {
                adjustMarks(marks, -1);
              } else {
                toggleMarksHidden(marks);
              }
              return;
            }

            const block = button.closest(".question, .answer");
            if (!block) {
              return;
            }

            if (action === "space-up") {
              adjustWorkingSpace(block, 8);
            } else if (action === "space-down") {
              adjustWorkingSpace(block, -8);
            } else if (action === "space-drag") {
              return;
            } else if (action === "toggle-split") {
              toggleSplit(block);
            } else if (action === "toggle-break") {
              toggleBreak(block);
            } else if (action === "edit-question") {
              openEditDialog(
                getQuestionEditableTarget(block),
                "Edit question",
              );
            } else if (action === "edit-answer") {
              openEditDialog(getAnswerEditableTarget(block), "Edit answer");
            } else if (action === "delete-question") {
              deleteQuestion(block);
            } else if (action === "move-up") {
              moveQuestion(block, -1);
            } else if (action === "move-down") {
              moveQuestion(block, 1);
            }
          });

          els.pageSize.addEventListener("change", () => {
            applyPageSizeFromSelect();
            applyPageSettings();
          });

          wireMarginInput("margin-top", "marginTopMm");
          wireMarginInput("margin-bottom", "marginBottomMm");
          wireMarginInput("margin-left", "marginLeftMm");
          wireMarginInput("margin-right", "marginRightMm");

          wireMarginsPopover();

          els.linesEnabled.addEventListener("change", applyLinesState);
          els.lineGap.addEventListener("input", () => {
            if (els.linesEnabled.checked) {
              applyLinesState();
            }
          });
          els.lineGap.addEventListener("change", () => {
            const v = clamp(Number(els.lineGap.value) || 8, 1, 50);
            els.lineGap.value = String(v);
            if (els.linesEnabled.checked) {
              applyLinesState();
            }
          });

          els.savePdf.addEventListener("click", savePdf);
          if (els.saveEdits) {
            els.saveEdits.addEventListener("click", openSaveEditsDialog);
          }
          window.addEventListener("pointermove", updateSpaceDrag);
          window.addEventListener("pointerup", endSpaceDrag);
          window.addEventListener("pointercancel", endSpaceDrag);
        }

        function startSpaceDrag(event, handle) {
          const block = handle.closest(".question");
          if (!block) {
            return;
          }

          event.preventDefault();
          state.resizeDrag = {
            block,
            handle,
            pointerId: event.pointerId,
            startY: event.clientY,
            startMm: readSpaceMm(block),
          };
          try {
            handle.setPointerCapture?.(event.pointerId);
          } catch {
            // Some synthetic pointer events cannot be captured.
          }
          handle.classList.add("is-dragging");
        }

        function updateSpaceDrag(event) {
          if (
            !state.resizeDrag ||
            event.pointerId !== state.resizeDrag.pointerId
          ) {
            return;
          }

          const deltaMm = (event.clientY - state.resizeDrag.startY) / MM_TO_PX;
          const next = clamp(state.resizeDrag.startMm + deltaMm, 0, 130);
          setWorkingSpace(state.resizeDrag.block, next);
        }

        function endSpaceDrag(event) {
          if (
            !state.resizeDrag ||
            event.pointerId !== state.resizeDrag.pointerId
          ) {
            return;
          }

          state.resizeDrag.handle.classList.remove("is-dragging");
          try {
            state.resizeDrag.handle.releasePointerCapture?.(event.pointerId);
          } catch {
            // Ignore release errors from already-ended pointer captures.
          }
          state.resizeDrag = null;
          scheduleLayout();
        }

        function adjustWorkingSpace(block, deltaMm) {
          if (block.classList.contains("page-break-after")) {
            return;
          }
          const current = readSpaceMm(block);
          setWorkingSpace(block, current + deltaMm);
        }

        function setWorkingSpace(block, nextMm) {
          block.dataset.workingSpaceMm = String(clamp(nextMm, 0, 130));
          syncQuestionSpace(block);
          scheduleLayout();
        }

        function toggleBreak(block) {
          const next = !block.classList.contains("page-break-after");
          block.classList.toggle("page-break-after", next);
          block.dataset.pageBreakAfter = next ? "true" : "false";

          const breakButton = block.querySelector(
            '[data-action="toggle-break"]',
          );
          if (breakButton) {
            breakButton.setAttribute("aria-pressed", String(next));
          }

          scheduleLayout();
        }

        function toggleSplit(block) {
          if (!block.classList.contains("question")) {
            return;
          }

          const next = !block.classList.contains("split-across-pages");
          block.classList.toggle("split-across-pages", next);
          block.dataset.allowWorkingSplit = next ? "true" : "false";

          const splitButton = block.querySelector(
            '[data-action="toggle-split"]',
          );
          if (splitButton) {
            splitButton.setAttribute("aria-pressed", String(next));
          }

          scheduleLayout();
        }

        function syncQuestionSpace(block) {
          const mm = readSpaceMm(block);
          const work = getPrimaryWorkingSpace(block);
          if (work) {
            work.style.height = `${mm}mm`;
          }
          const value = block.querySelector("[data-space-value]");
          if (value) {
            value.textContent = `${Math.round(mm)} mm`;
          }
          positionQuestionTools(block);
        }

        function readSpaceMm(block) {
          const raw = Number(block.dataset.workingSpaceMm);
          return Number.isFinite(raw) ? raw : 34;
        }

        function applyPageSettings() {
          const page = getPage();
          const printableWidthPx = Math.max(
            260,
            (page.width - state.marginLeftMm - state.marginRightMm) *
              MM_TO_PX,
          );
          const printableHeightPx = Math.max(
            260,
            (page.height - state.marginTopMm - state.marginBottomMm) *
              MM_TO_PX,
          );

          els.root.style.setProperty(
            "--page-width-px",
            `${page.width * MM_TO_PX}px`,
          );
          els.root.style.setProperty(
            "--page-height-px",
            `${page.height * MM_TO_PX}px`,
          );
          els.root.style.setProperty(
            "--page-margin-top-px",
            `${state.marginTopMm * MM_TO_PX}px`,
          );
          els.root.style.setProperty(
            "--page-margin-bottom-px",
            `${state.marginBottomMm * MM_TO_PX}px`,
          );
          els.root.style.setProperty(
            "--page-margin-left-px",
            `${state.marginLeftMm * MM_TO_PX}px`,
          );
          els.root.style.setProperty(
            "--page-margin-right-px",
            `${state.marginRightMm * MM_TO_PX}px`,
          );
          els.root.style.setProperty(
            "--page-gap-px",
            `${SCREEN_PAGE_GAP_PX}px`,
          );
          els.root.style.setProperty(
            "--printable-width-px",
            `${printableWidthPx}px`,
          );
          els.root.style.setProperty(
            "--printable-height-px",
            `${printableHeightPx}px`,
          );
          els.root.style.setProperty(
            "--rule-gap-px",
            `${Math.max(1, state.lineGapMm * MM_TO_PX)}px`,
          );

          els.printStyle.textContent = `@page { size: ${page.width}mm ${page.height}mm; margin: ${state.marginTopMm}mm ${state.marginRightMm}mm ${state.marginBottomMm}mm ${state.marginLeftMm}mm; }`;

          updateMarginsSummary();
          scheduleLayout();
        }

        function updateMarginsSummary() {
          const summary = document.getElementById("margins-summary");
          if (!summary) return;
          summary.textContent = `${state.marginTopMm}·${state.marginRightMm}·${state.marginBottomMm}·${state.marginLeftMm} mm`;
        }

        function applyPageSizeFromSelect() {
          const raw = els.pageSize.value || "a4-portrait";
          const idx = raw.lastIndexOf("-");
          const size = idx > 0 ? raw.slice(0, idx) : raw;
          const orient = idx > 0 ? raw.slice(idx + 1) : "portrait";
          if (PAGE_SIZES[size]) {
            state.pageSize = size;
          }
          if (orient === "portrait" || orient === "landscape") {
            state.orientation = orient;
          }
        }

        function syncPageSizeSelect() {
          if (!els.pageSize) return;
          const target = `${state.pageSize}-${state.orientation}`;
          if (
            Array.from(els.pageSize.options).some(
              (opt) => opt.value === target,
            )
          ) {
            els.pageSize.value = target;
          }
        }

        function wireMarginInput(inputId, stateKey) {
          const input = document.getElementById(inputId);
          if (!input) return;
          const apply = () => {
            const v = clamp(Number(input.value) || 0, 0, 50);
            state[stateKey] = v;
            input.value = String(v);
            applyPageSettings();
          };
          input.addEventListener("input", () => {
            const v = clamp(Number(input.value) || 0, 0, 50);
            state[stateKey] = v;
            applyPageSettings();
          });
          input.addEventListener("change", apply);
          input.addEventListener("blur", apply);
        }

        function wireMarginsPopover() {
          const field = document.getElementById("margins-field");
          const trigger = document.getElementById("margins-trigger");
          const popover = document.getElementById("margins-popover");
          if (!field || !trigger || !popover) return;

          let hideTimer = 0;
          const cancelHide = () => {
            if (hideTimer) {
              clearTimeout(hideTimer);
              hideTimer = 0;
            }
          };
          const open = () => {
            cancelHide();
            field.classList.add("is-open");
            trigger.setAttribute("aria-expanded", "true");
          };
          const scheduleClose = () => {
            cancelHide();
            hideTimer = setTimeout(() => {
              field.classList.remove("is-open");
              trigger.setAttribute("aria-expanded", "false");
              hideTimer = 0;
            }, 1000);
          };

          [trigger, popover].forEach((el) => {
            el.addEventListener("pointerenter", open);
            el.addEventListener("pointerleave", scheduleClose);
          });
          trigger.addEventListener("focus", open);
          popover.addEventListener("focusin", open);
          popover.addEventListener("focusout", scheduleClose);
          trigger.addEventListener("click", () => {
            if (field.classList.contains("is-open")) {
              cancelHide();
              field.classList.remove("is-open");
              trigger.setAttribute("aria-expanded", "false");
            } else {
              open();
            }
          });
        }

        function syncMarginInputs() {
          const set = (id, value) => {
            const input = document.getElementById(id);
            if (input) input.value = String(value);
          };
          set("margin-top", state.marginTopMm);
          set("margin-bottom", state.marginBottomMm);
          set("margin-left", state.marginLeftMm);
          set("margin-right", state.marginRightMm);
        }

        function applyLinesState() {
          if (els.linesEnabled.checked) {
            const v = clamp(Number(els.lineGap.value) || 8, 1, 50);
            state.lineGapMm = v;
            els.body.classList.remove("no-ruled-lines");
          } else {
            state.lineGapMm = 0;
            els.body.classList.add("no-ruled-lines");
          }
          els.root.style.setProperty(
            "--rule-gap-px",
            `${Math.max(1, state.lineGapMm * MM_TO_PX)}px`,
          );
          scheduleLayout();
        }

        function getPage() {
          const base = PAGE_SIZES[state.pageSize];
          const portrait = state.orientation === "portrait";
          const width = portrait ? base.width : base.height;
          const height = portrait ? base.height : base.width;
          return {
            width,
            height,
            label: base.label,
            orientation: state.orientation,
          };
        }

        function scheduleLayout() {
          cancelAnimationFrame(state.layoutRaf);
          state.layoutRaf = requestAnimationFrame(updateScreenPagination);
        }

        function updateScreenPagination() {
          const printableHeight = getLayoutHeightPx();
          const blocks = Array.from(
            els.content.querySelectorAll(".question, .answer"),
          );

          els.content.querySelectorAll(".starts-new-page").forEach((item) => {
            item.classList.remove("starts-new-page");
          });
          els.content
            .querySelectorAll(".natural-page-spacer")
            .forEach((spacer) => spacer.remove());
          els.content
            .querySelectorAll(".working-space-continuation")
            .forEach((space) => space.remove());
          els.content.querySelectorAll(".question").forEach((block) => {
            syncQuestionSpace(block);
          });
          blocks.forEach((block) => {
            const spacer = block.querySelector(":scope > .manual-break-spacer");
            if (spacer) {
              spacer.style.setProperty("--page-fill-px", "0px");
              spacer.style.height = "0px";
            }
          });

          void els.content.offsetHeight;
          positionQuestionTools();

          let pageY = 0;
          let pages = 1;
          let nextStartsNewPage = false;
          const items = Array.from(els.content.children).filter((item) => {
            return !item.classList.contains("natural-page-spacer");
          });

          items.forEach((item, index) => {
            if (item.classList.contains("manual-break-spacer")) {
              return;
            }

            if (nextStartsNewPage) {
              item.classList.add("starts-new-page");
              nextStartsNewPage = false;
            }

            const itemHeight = getOuterHeight(item);
            const itemContentHeight = getBoxHeight(item);
            const nextItem = items[index + 1];
            const keepWithNextHeight = shouldKeepWithNext(item, nextItem)
              ? getOuterHeight(nextItem)
              : 0;
            const fitHeight = itemHeight + keepWithNextHeight;
            const canTrySplit = isSplittableQuestion(item);
            const fitsCurrentPage =
              pageY + fitHeight <= printableHeight ||
              (pageY === 0 && !canTrySplit);
            const canMoveWholeItem = fitHeight < printableHeight * 0.98;

            if (!fitsCurrentPage) {
              if (canTrySplit && pageY + itemContentHeight <= printableHeight) {
                pageY += itemContentHeight;

                if (item.classList.contains("page-break-after")) {
                  const spacer = item.querySelector(
                    ":scope > .manual-break-spacer",
                  );
                  setPageSpacerHeight(
                    spacer,
                    Math.max(
                      0,
                      printableHeight - Math.min(pageY, printableHeight),
                    ),
                  );
                  pages += 1;
                  pageY = 0;
                  nextStartsNewPage = true;
                }
                return;
              }

              const split = splitQuestionWorkingSpace(
                item,
                pageY,
                printableHeight,
              );
              if (split) {
                pages += 1;
                pageY = split.pageY;

                if (item.classList.contains("page-break-after")) {
                  const spacer = item.querySelector(
                    ":scope > .manual-break-spacer",
                  );
                  setPageSpacerHeight(
                    spacer,
                    Math.max(
                      0,
                      printableHeight - Math.min(pageY, printableHeight),
                    ),
                  );
                  pages += 1;
                  pageY = 0;
                  nextStartsNewPage = true;
                } else if (pageY > printableHeight) {
                  const overflowPages = Math.floor(pageY / printableHeight);
                  pages += overflowPages;
                  pageY = pageY % printableHeight;
                }
                return;
              }

              if (!canMoveWholeItem) {
                pageY += itemHeight;
                if (pageY > printableHeight) {
                  const overflowPages = Math.floor(pageY / printableHeight);
                  pages += overflowPages;
                  pageY = pageY % printableHeight;
                }
                return;
              }

              const naturalSpacer = createPageSpacer(printableHeight - pageY);
              els.content.insertBefore(naturalSpacer, item);
              item.classList.add("starts-new-page");
              pages += 1;
              pageY = 0;
            }

            pageY += itemHeight;

            if (item.classList.contains("page-break-after")) {
              const spacer = item.querySelector(
                ":scope > .manual-break-spacer",
              );
              setPageSpacerHeight(
                spacer,
                Math.max(0, printableHeight - Math.min(pageY, printableHeight)),
              );
              pages += 1;
              pageY = 0;
              nextStartsNewPage = true;
              return;
            }

            if (pageY > printableHeight) {
              const overflowPages = Math.floor(pageY / printableHeight);
              pages += overflowPages;
              pageY = pageY % printableHeight;
            }
          });

          state.renderedPageCount = pages;
          positionQuestionTools();
          positionEmbedMarkers();
          updateMoveButtonStates();
          updatePageCount();
        }

        function createPageSpacer(fillPx) {
          const spacer = document.createElement("div");
          spacer.className = "screen-page-spacer natural-page-spacer no-pdf";
          spacer.setAttribute("aria-hidden", "true");
          setPageSpacerHeight(spacer, fillPx);
          return spacer;
        }

        function splitQuestionWorkingSpace(item, pageY, printableHeight) {
          if (!isSplittableQuestion(item)) {
            return null;
          }

          const work = getPrimaryWorkingSpace(item);
          if (!work) {
            return null;
          }

          const promptAndGapHeight = work.offsetTop;
          const workingHeight = work.getBoundingClientRect().height;
          const available = printableHeight - pageY;

          if (
            available <= promptAndGapHeight + MIN_SPLIT_WORKING_PX ||
            available >= promptAndGapHeight + workingHeight - 1
          ) {
            return null;
          }

          const firstWorkingHeight = Math.max(
            MIN_SPLIT_WORKING_PX,
            available - promptAndGapHeight,
          );
          const remainingWorkingHeight = Math.max(
            MIN_SPLIT_WORKING_PX,
            workingHeight - firstWorkingHeight,
          );

          work.style.height = `${firstWorkingHeight}px`;

          const spacer = createPageSpacer(0);
          const continuation = document.createElement("div");
          continuation.className =
            "working-space working-space-continuation starts-new-page";
          continuation.setAttribute("aria-hidden", "true");
          continuation.style.height = `${remainingWorkingHeight}px`;

          work.insertAdjacentElement("afterend", spacer);
          spacer.insertAdjacentElement("afterend", continuation);

          void item.offsetHeight;
          positionQuestionTools(item);

          const style = window.getComputedStyle(item);
          const itemMarginBottom = parseFloat(style.marginBottom) || 0;
          return {
            pageY: getOuterHeight(continuation) + itemMarginBottom,
          };
        }

        function setPageSpacerHeight(spacer, fillPx) {
          if (!spacer) {
            return;
          }
          const fill = Math.max(0, fillPx);
          const total =
            fill +
            (state.marginTopMm + state.marginBottomMm) * MM_TO_PX +
            SCREEN_PAGE_GAP_PX;
          spacer.style.setProperty("--page-fill-px", `${fill}px`);
          spacer.style.height = `${total}px`;
        }

        function getOuterHeight(element) {
          const rect = element.getBoundingClientRect();
          const style = window.getComputedStyle(element);
          const marginTop = parseFloat(style.marginTop) || 0;
          const marginBottom = parseFloat(style.marginBottom) || 0;
          return rect.height + marginTop + marginBottom;
        }

        function getBoxHeight(element) {
          return element.getBoundingClientRect().height;
        }

        function getPrimaryWorkingSpace(block) {
          return block.querySelector(
            ":scope > .working-space:not(.working-space-continuation)",
          );
        }

        function isSplittableQuestion(item) {
          return (
            item.classList.contains("question") &&
            item.classList.contains("split-across-pages")
          );
        }

        function positionQuestionTools(block) {
          const questions = block
            ? [block]
            : Array.from(els.content.querySelectorAll(".question"));

          questions.forEach((question) => {
            const tools = question.querySelector(
              ":scope > .pagination-tools--spacing",
            );
            const spaces = Array.from(
              question.querySelectorAll(":scope > .working-space"),
            );
            const target = spaces[spaces.length - 1];
            if (!tools || !target) {
              return;
            }

            const top = Math.max(
              0,
              target.offsetTop + target.offsetHeight - tools.offsetHeight,
            );
            question.style.setProperty("--question-tools-top", `${top}px`);
          });
        }

        function shouldKeepWithNext(item, nextItem) {
          if (!nextItem) {
            return false;
          }
          const isHeadingBlock =
            item.classList.contains("worksheet-section") ||
            item.classList.contains("answers-title");
          const isContentBlock =
            nextItem.classList.contains("question") ||
            nextItem.classList.contains("answer");
          return isHeadingBlock && isContentBlock;
        }

        function updatePageCount() {
          const pages = Math.max(1, state.renderedPageCount || 1);
          els.pageCount.textContent = `${pages} ${pages === 1 ? "page" : "pages"}`;
        }

        function getLayoutHeightPx() {
          return Math.max(1, getPrintableHeightPx() - PAGE_BREAK_SAFETY_PX);
        }

        function getPrintableHeightPx() {
          const page = getPage();
          return Math.max(
            260,
            (page.height - state.marginTopMm - state.marginBottomMm) *
              MM_TO_PX,
          );
        }

        // html2canvas only partially supports inline SVG (clip paths, text
        // anchors), so diagrams and rendered graphs are swapped for 2x PNG
        // snapshots for the duration of the export.
        async function rasterizeSvgsForExport(root) {
          const svgs = Array.from(
            root.querySelectorAll(".fn-graph svg, svg.worksheet-diagram"),
          );
          const swaps = [];
          await Promise.all(
            svgs.map(async (svg) => {
              try {
                const img = await svgToPngImage(svg);
                if (!img) return;
                svg.replaceWith(img);
                swaps.push({ svg, img });
              } catch (err) {
                console.warn("SVG rasterize failed; exporting as-is:", err);
              }
            }),
          );
          return () => {
            swaps.forEach(({ svg, img }) => {
              img.replaceWith(svg);
            });
          };
        }

        async function svgToPngImage(svg) {
          const rect = svg.getBoundingClientRect();
          const width = Math.max(1, Math.round(rect.width));
          const height = Math.max(1, Math.round(rect.height));
          const clone = svg.cloneNode(true);
          clone.setAttribute("width", String(width));
          clone.setAttribute("height", String(height));
          clone.style.removeProperty("width");
          clone.style.removeProperty("height");
          if (!clone.getAttribute("xmlns")) {
            clone.setAttribute("xmlns", "http://www.w3.org/2000/svg");
          }
          inlineSvgComputedStyles(svg, clone);
          const markup = new XMLSerializer().serializeToString(clone);
          const blob = new Blob([markup], {
            type: "image/svg+xml;charset=utf-8",
          });
          const url = URL.createObjectURL(blob);
          try {
            const image = await loadImageFromUrl(url);
            const canvas = document.createElement("canvas");
            canvas.width = width * 2;
            canvas.height = height * 2;
            const ctx = canvas.getContext("2d");
            ctx.scale(2, 2);
            ctx.drawImage(image, 0, 0, width, height);
            const img = document.createElement("img");
            img.src = canvas.toDataURL("image/png");
            img.width = width;
            img.height = height;
            img.alt = svg.getAttribute("aria-label") || "";
            return img;
          } finally {
            URL.revokeObjectURL(url);
          }
        }

        function loadImageFromUrl(url) {
          return new Promise((resolve, reject) => {
            const image = new Image();
            image.onload = () => resolve(image);
            image.onerror = () =>
              reject(new Error("SVG image failed to load"));
            image.src = url;
          });
        }

        function inlineSvgComputedStyles(source, clone) {
          const props = [
            "fill",
            "stroke",
            "stroke-width",
            "stroke-dasharray",
            "stroke-linecap",
            "stroke-linejoin",
            "opacity",
            "fill-opacity",
            "stroke-opacity",
            "font-family",
            "font-size",
            "font-weight",
            "font-style",
            "text-anchor",
            "dominant-baseline",
            "visibility",
          ];
          const sourceEls = [source, ...source.querySelectorAll("*")];
          const cloneEls = [clone, ...clone.querySelectorAll("*")];
          sourceEls.forEach((el, i) => {
            const copy = cloneEls[i];
            if (!copy || !(el instanceof Element)) return;
            const computed = window.getComputedStyle(el);
            props.forEach((prop) => {
              const value = computed.getPropertyValue(prop);
              if (value) {
                copy.style.setProperty(prop, value);
              }
            });
          });
        }

        async function savePdf() {
          const page = getPage();
          const filename = makeFilename();

          els.savePdf.disabled = true;
          els.savePdfLabel.textContent = "Saving";

          let restoreSvgs = null;
          try {
            updateScreenPagination();
            els.body.classList.add("exporting");
            await nextFrame();
            restoreSvgs = await rasterizeSvgsForExport(els.content);
            await nextFrame();
            await waitForPdfLibrary();

            await window
              .html2pdf()
              .set({
                margin: [
                  state.marginTopMm,
                  state.marginRightMm,
                  state.marginBottomMm,
                  state.marginLeftMm,
                ],
                filename,
                image: { type: "png", quality: 1 },
                html2canvas: {
                  scale: 2,
                  useCORS: true,
                  backgroundColor: "#ffffff",
                  scrollX: 0,
                  scrollY: 0,
                  windowWidth: document.documentElement.clientWidth,
                },
                jsPDF: {
                  unit: "mm",
                  format: [page.width, page.height],
                  orientation: page.orientation,
                },
                pagebreak: {
                  mode: ["css", "legacy"],
                  before: [".starts-new-page"],
                  avoid: [
                    ".question:not(.split-across-pages)",
                    ".answer",
                    ".worksheet-section",
                    ".answers-title",
                    ".question-prompt",
                    ".answer-body",
                    ".fn-graph",
                  ],
                },
              })
              .from(els.content)
              .save();
          } catch (error) {
            console.error(error);
            showMessage({
              title: "PDF export failed",
              message:
                "The PDF library did not finish correctly. No browser print fallback was used, so headers and footers will not be added by this button.",
              confirmLabel: "OK",
            });
          } finally {
            if (restoreSvgs) {
              restoreSvgs();
            }
            els.body.classList.remove("exporting");
            els.savePdf.disabled = false;
            els.savePdfLabel.textContent = "Save PDF";
            scheduleLayout();
          }
        }

        function renderMath() {
          const run = () => {
            if (typeof window.renderMathInElement === "function") {
              window.renderMathInElement(els.content, {
                delimiters: KATEX_DELIMITERS,
                output: "html",
                throwOnError: false,
              });
              scheduleLayout();
            }
          };

          if (document.readyState === "complete") {
            run();
          } else {
            window.addEventListener("load", run, { once: true });
          }
        }

        function makeFilename() {
          const title = els.content.querySelector("h1");
          const raw = title ? title.textContent : "worksheet";
          const slug = raw
            .trim()
            .toLowerCase()
            .replace(/[^a-z0-9]+/g, "-")
            .replace(/^-|-$/g, "");
          return `${slug || "worksheet"}-pagination-demo.pdf`;
        }

        function nextFrame() {
          return new Promise((resolve) =>
            requestAnimationFrame(() => resolve()),
          );
        }

        async function waitForPdfLibrary() {
          for (let attempt = 0; attempt < 50; attempt += 1) {
            if (typeof window.html2pdf === "function") {
              return;
            }
            await new Promise((resolve) => setTimeout(resolve, 100));
          }
          throw new Error("html2pdf did not load");
        }

        function clamp(value, min, max) {
          return Math.min(max, Math.max(min, value));
        }

        function loadStateFromBody() {
          const body = document.body;
          const data = body.dataset;
          if (data.savedPageSize && PAGE_SIZES[data.savedPageSize]) {
            state.pageSize = data.savedPageSize;
          }
          if (
            data.savedOrientation === "portrait" ||
            data.savedOrientation === "landscape"
          ) {
            state.orientation = data.savedOrientation;
          }
          const parseMargin = (raw) => {
            const v = Number(raw);
            return Number.isFinite(v) ? clamp(v, 0, 50) : null;
          };
          const legacyY = parseMargin(data.savedMarginYMm);
          const legacyX = parseMargin(data.savedMarginXMm);
          const top = parseMargin(data.savedMarginTopMm) ?? legacyY;
          const bottom = parseMargin(data.savedMarginBottomMm) ?? legacyY;
          const left = parseMargin(data.savedMarginLeftMm) ?? legacyX;
          const right = parseMargin(data.savedMarginRightMm) ?? legacyX;
          if (top !== null) state.marginTopMm = top;
          if (bottom !== null) state.marginBottomMm = bottom;
          if (left !== null) state.marginLeftMm = left;
          if (right !== null) state.marginRightMm = right;
          if (data.savedLineGapMm !== undefined) {
            const g = Number(data.savedLineGapMm);
            if (Number.isFinite(g) && g >= 0) state.lineGapMm = g;
          }
          const preferredRaw = Number(data.savedLinePreferredMm);
          const preferredLine = Number.isFinite(preferredRaw)
            ? clamp(preferredRaw, 1, 50)
            : state.lineGapMm > 0
              ? state.lineGapMm
              : 8;

          syncPageSizeSelect();
          syncMarginInputs();

          els.linesEnabled.checked = state.lineGapMm > 0;
          els.lineGap.value = String(preferredLine);

          if (state.lineGapMm === 0) {
            els.body.classList.add("no-ruled-lines");
          } else {
            els.body.classList.remove("no-ruled-lines");
          }

          updateMarginsSummary();
        }

        function ensureSaveDialog() {
          let dialog = document.getElementById("save-edits-dialog");
          if (dialog) {
            return dialog;
          }
          dialog = document.createElement("dialog");
          dialog.id = "save-edits-dialog";
          dialog.className = "no-pdf";
          dialog.innerHTML = `
            <form method="dialog" class="edit-dialog-form">
              <h3>Save edited worksheet</h3>
              <p class="edit-dialog-hint">A standalone HTML copy will be downloaded with your current margins, breaks, marks, and edits baked in. Reopen it in any browser to keep going.</p>
              <label class="save-filename-field" for="save-edits-filename">
                Filename
                <span class="save-filename-row">
                  <input id="save-edits-filename" type="text" autocomplete="off" spellcheck="false" />
                  <span class="save-extension">.html</span>
                </span>
              </label>
              <div class="edit-dialog-actions">
                <button type="button" data-action="save-dialog-cancel">Cancel</button>
                <button type="submit" value="ok" data-action="save-dialog-ok">Download</button>
              </div>
            </form>
          `;
          document.body.appendChild(dialog);

          dialog
            .querySelector('[data-action="save-dialog-cancel"]')
            .addEventListener("click", () => {
              dialog.close("cancel");
            });

          dialog.addEventListener("close", () => {
            if (dialog.returnValue !== "ok") {
              return;
            }
            const input = dialog.querySelector("#save-edits-filename");
            const raw = (input.value || "worksheet").trim();
            const cleaned = raw
              .replace(/\.html?$/i, "")
              .replace(/[<>:"/\\|?*]+/g, "-")
              .replace(/\s+/g, "-")
              .replace(/-+/g, "-")
              .replace(/^-|-$/g, "");
            const filename = (cleaned || "worksheet") + ".html";
            downloadEditedHtml(filename);
          });

          return dialog;
        }

        function openSaveEditsDialog() {
          const dialog = ensureSaveDialog();
          const input = dialog.querySelector("#save-edits-filename");
          const base = makeFilename()
            .replace(/\.pdf$/i, "")
            .replace(/-pagination-demo$/i, "");
          input.value = base || "worksheet";
          dialog.returnValue = "";
          dialog.showModal();
          requestAnimationFrame(() => {
            input.focus();
            input.select();
          });
        }

        function downloadEditedHtml(filename) {
          const html = serializeForSave();
          const blob = new Blob([html], {
            type: "text/html;charset=utf-8",
          });
          const url = URL.createObjectURL(blob);
          const a = document.createElement("a");
          a.href = url;
          a.download = filename;
          a.style.display = "none";
          document.body.appendChild(a);
          a.click();
          setTimeout(() => {
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
          }, 0);
        }

        function serializeForSave() {
          const clone = document.documentElement.cloneNode(true);

          const ephemeralSelectors = [
            "#question-edit-dialog",
            "#save-edits-dialog",
            "#message-dialog",
            "#latex-edit-dialog",
            "#image-insert-dialog",
            "#graph-editor-dialog",
            "#diagram-editor-dialog",
            "#dialog-backdrop",
            ".screen-page-spacer",
            ".manual-break-spacer",
            ".working-space",
            ".pagination-tools",
            ".content-element-tools",
          ];
          ephemeralSelectors.forEach((sel) => {
            clone.querySelectorAll(sel).forEach((n) => n.remove());
          });

          // Rendered plots are ephemeral: keep only the data-graph spec and
          // re-render when the saved file is opened.
          clone.querySelectorAll(".fn-graph").forEach((n) => {
            n.innerHTML = "";
            n.removeAttribute("data-graph-error");
          });

          clone.querySelectorAll(".starts-new-page").forEach((n) => {
            n.classList.remove("starts-new-page");
            if (n.classList.length === 0) {
              n.removeAttribute("class");
            }
          });

          clone.querySelectorAll(".editable-text-target").forEach((n) => {
            n.classList.remove("editable-text-target");
            if (n.classList.length === 0) {
              n.removeAttribute("class");
            }
          });

          clone.querySelectorAll(".question").forEach((n) => {
            n.style.removeProperty("--question-tools-top");
            if (n.getAttribute("style") === "") {
              n.removeAttribute("style");
            }
          });

          clone.querySelectorAll(".marks").forEach((marks) => {
            let value = Number(marks.getAttribute("data-marks-value"));
            if (!Number.isFinite(value) || value < 1) {
              const m = (marks.textContent || "").match(/(\d+)/);
              value = m ? Math.max(1, parseInt(m[1], 10)) : 1;
            }
            const hidden = marks.classList.contains("marks-hidden");
            marks.removeAttribute("data-marks-decorated");
            marks.removeAttribute("data-marks-value");
            marks.className = hidden ? "marks marks-hidden" : "marks";
            marks.textContent = `${value} ${value === 1 ? "mark" : "marks"}`;
          });

          const body = clone.querySelector("body");
          body.dataset.savedPageSize = state.pageSize;
          body.dataset.savedOrientation = state.orientation;
          body.dataset.savedMarginTopMm = String(state.marginTopMm);
          body.dataset.savedMarginBottomMm = String(state.marginBottomMm);
          body.dataset.savedMarginLeftMm = String(state.marginLeftMm);
          body.dataset.savedMarginRightMm = String(state.marginRightMm);
          body.dataset.savedLineGapMm = String(state.lineGapMm);
          const preferredLineRaw = clamp(
            Number(els.lineGap.value) ||
              (state.lineGapMm > 0 ? state.lineGapMm : 8),
            1,
            50,
          );
          body.dataset.savedLinePreferredMm = String(preferredLineRaw);
          delete body.dataset.savedMarginYMm;
          delete body.dataset.savedMarginXMm;
          body.classList.remove("exporting");
          body.classList.remove("hide-guides");
          if (state.lineGapMm === 0) {
            body.classList.add("no-ruled-lines");
          } else {
            body.classList.remove("no-ruled-lines");
          }

          const setSelected = (selectEl, value) => {
            if (!selectEl) return;
            selectEl.querySelectorAll("option").forEach((opt) => {
              if (opt.value === value) {
                opt.setAttribute("selected", "");
              } else {
                opt.removeAttribute("selected");
              }
            });
          };

          setSelected(
            clone.querySelector("#page-size"),
            `${state.pageSize}-${state.orientation}`,
          );

          const lineGapInput = clone.querySelector("#line-gap");
          if (lineGapInput) {
            lineGapInput.setAttribute("value", String(preferredLineRaw));
          }
          const linesEnabled = clone.querySelector("#lines-enabled");
          if (linesEnabled) {
            if (state.lineGapMm > 0) {
              linesEnabled.setAttribute("checked", "");
            } else {
              linesEnabled.removeAttribute("checked");
            }
          }

          const setMarginValue = (id, value) => {
            const el = clone.querySelector(id);
            if (el) el.setAttribute("value", String(value));
          };
          setMarginValue("#margin-top", state.marginTopMm);
          setMarginValue("#margin-bottom", state.marginBottomMm);
          setMarginValue("#margin-left", state.marginLeftMm);
          setMarginValue("#margin-right", state.marginRightMm);

          const summary = clone.querySelector("#margins-summary");
          if (summary) {
            summary.textContent = `${state.marginTopMm}·${state.marginRightMm}·${state.marginBottomMm}·${state.marginLeftMm} mm`;
          }

          return "<!doctype html>\n" + clone.outerHTML;
        }
      })();
    </script>
